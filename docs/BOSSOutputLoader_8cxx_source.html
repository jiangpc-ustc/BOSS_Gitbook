26  BOSSOutputLoader::BOSSOutputLoader(const char* path, bool print) :
27  fDirectoryPath(path)
28  {
31  if(fDirectoryPath.EndsWith(".root")) {
34  TFile file(fDirectoryPath.Data());
35  if(file.IsZombie()) return;
37  TString fname(fDirectoryPath);
38  auto pos = fDirectoryPath.Last('/');
39  fDirectoryPath.Resize(pos);
40  fname.Remove(0, pos+1);
42  fFileNames.push_back(fname);
44  }
46  else {
49  if(fDirectoryPath.EndsWith("/")) fDirectoryPath.Chop();
51  TSystemDirectory dir(path, path);
53  TList *files = dir.GetListOfFiles();
54  if(!files) {
55  std::cout << "ERROR: Directory \"" << fDirectoryPath << "\" does not exist" << std::endl;
56  return;
57  }
59  TIter fileIter(files);
60  TObject *obj;
61  while ((obj = fileIter())) {
62  auto systemFile = dynamic_cast<TSystemFile*>(obj);
63  if(!systemFile) continue;
64  TString filename { systemFile->GetName() };
65  if(!filename.EndsWith(".root")) continue;
66  TString fname(systemFile->GetName());
68  TFile file((fDirectoryPath+"/"+fname).Data());
69  if(!file.IsZombie()) fFileNames.push_back(fname);
70  }
72  }
74  LoadChains(print);
75  }

383  std::list<ChainLoader*> BOSSOutputLoader::CreateOrderedMap()
384  {
385  // * Create list of pointers to `ChainLoader` * //
386  std::list<ChainLoader*> outputList;
387  for(auto it = fChains.begin(); it != fChains.end(); ++it) {
388  TString name(it->second.GetChain().GetName());
389  if(!name.EqualTo("_cutvalues")) outputList.push_back(&(it->second));
390  }
391  // * Sort resulting list based on number of entries * //
392  outputList.sort([](ChainLoader* const & a, ChainLoader* const & b)
393  {
394  std::string nameA(a->GetChain().GetName());
395  std::string nameB(b->GetChain().GetName());
396  return a->GetEntries() != b->GetEntries() ?
397  a->GetEntries() > b->GetEntries() :
398  nameA < nameB;
399  });
400  return outputList;
