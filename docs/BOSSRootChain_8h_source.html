305 void BOSSRootChain::LoadChains()
306 {
307  // * Attempt to load one of the files *
308  if(!fFileNames.size()) {
309  std::cout << "ERROR: List of file names is empty!" << std::endl;
310  std::cout << "--> There were no ROOT files in directory \"" << fDirectoryPath << " satisfying the \"" << fAnalysisName << "\" pattern" << std::endl;
311  }
312  TString fileToLoad(fDirectoryPath+"/"+fFileNames.front());
313  TFile file(fileToLoad.Data());
314  if(file.IsZombie()) {
315  std::cout << "ERROR: File \"" << fileToLoad << "\" is zombie" << std::endl;
316  }
317  // * Go through keys in the file and add create TChain if it is a TTree *
318  TIter next(file.GetListOfKeys());
319  TObject *obj = nullptr;
320  TKey *key = nullptr;
321  TTree *tree = nullptr;
322  while((obj = next())) {
323  if((key = dynamic_cast<TKey*>(obj))) { // attempt to cast to TKey
324  if((tree = dynamic_cast<TTree*>(key->ReadObj()))) { // read TKey and attempt to cast to TTree
325  TChain* chain = new TChain(tree->GetName());
326  // * Loop over file names and at them to the chain
327  std::list<TString>::iterator it = fFileNames.begin();
328  while(it != fFileNames.end()) {
329  chain->Add(fDirectoryPath+"/"+(*it)); // add ROOT file
330  ++it;
331  }
332  // * Add chain pointer to the list
333  fChains.push_back(chain);
334  }
335  }
336  }
337  // * Output *
338  std::cout
339  << "Created a list of " << fChains.size() << " TChains based on file \"" << file.GetName() << "\""
340  << " with " << fFileNames.size() << " ROOT files linked to them" << std::endl;
341 }
