23  void ChainLoader::BookAddresses(bool print)
24  {
25  if(!fChain.GetNbranches()) return;
27  TIter next(fChain.GetListOfBranches());
28  TObject *obj = nullptr;
29  if(print) {
30  std::cout << "Tree \"" << fChain.GetName() << "\" (" << fChain.GetEntries() << " entries)" << std::endl;
31  std::cout << fChain.GetTitle() << std::endl;
32  std::cout << " "
33  << std::setw(18) << std::left << "BRANCH NAME"
34  << std::setw(12) << std::right << "MEAN" << std::endl;
35  }
37  while((obj = next())) {
38  std::string type(obj->GetTitle());
39  if(print) std::cout << " "
40  << std::setw(18) << std::left << type
41  << std::setw(12) << std::right << ComputeMean(&fChain, obj->GetName()) << std::endl;
42  switch(type.back()) {
43  case 'B' : SetAddress(obj, fMap_B); break;
44  case 'b' : SetAddress(obj, fMap_b); break;
45  case 'S' : SetAddress(obj, fMap_S); break;
46  case 's' : SetAddress(obj, fMap_s); break;
47  case 'I' : SetAddress(obj, fMap_I); break;
48  case 'i' : SetAddress(obj, fMap_i); break;
49  case 'F' : SetAddress(obj, fMap_F); break;
50  case 'D' : SetAddress(obj, fMap_D); break;
51  case 'L' : SetAddress(obj, fMap_L); break;
52  case 'l' : SetAddress(obj, fMap_l); break;
53  case 'O' : SetAddress(obj, fMap_O); break;
54  default :
55  std::cout << "ERROR: Unable to book address for branch \"" << type << "\"" << std::endl;
56  }
57  }
58  if(print) std::cout << std::endl;
