137  void FitPureDoubleGaussian(TH1D &hist, const ReconstructedParticle& particle)
138  {
139 
140  // * The `RooFit` method * //
141  RooRealVar invMassVar("invMassVar",
142  Form("#it{M}_{%s} (GeV/#it{c}^{2})", particle.GetDaughterLabel()),
143  particle.PlotFrom(),
144  particle.PlotUntil()
145  );
146 
147  // * Import histogram to Roofit * //
148  RooDataHist invMassDistribution(
149  Form("%scandidate_RooDataHist", particle.GetName()), hist.GetTitle(),
150  invMassVar, RooFit::Import(hist));
151 
152  // * Create Gaussian functions * //
153  RooRealVar mean(
154  Form("m_{%s}", particle.GetNameLaTeX()),
155  Form("%s mass", particle.GetNameLaTeX()),
156  particle.GetMass(), particle.GetLowerMass(), particle.GetUpperMass());
157  RooRealVar s1("#sigma_{1}", Form("%s width 1", particle.GetNameLaTeX()),
158  particle.GetGaussianSmallWidth(), 0., 10.*particle.GetGaussianSmallWidth());
159  RooRealVar s2("#sigma_{2}", Form("%s width 2", particle.GetNameLaTeX()),
160  particle.GetGaussianWideWidth(), 0., 10.*particle.GetGaussianWideWidth());
161  RooGaussian gauss1("gauss1",
162  "Gaussian PDF 1 for #it{M}_{#gamma#gamma} distribution",
163  invMassVar, mean, s1);
164  RooGaussian gauss2("gauss2",
165  "Gaussian PDF 2 for #it{M}_{#gamma#gamma} distribution",
166  invMassVar, mean, s2);
167 
168  // * Add the components and fit * //
169  RooRealVar ratio("n_{gaus1} / n_{gaus2}", "Ratio between the two Gaussian pdfs", .8, 0., 1.);
170  RooAddPdf doublegauss("double_gaussian", "Double gaussian",
171  RooArgList(gauss1, gauss2), RooArgList(ratio));
172  doublegauss.fitTo(
173  invMassDistribution,
174  RooFit::Range(particle.FitFrom(), particle.FitUntil()));
175 
176  // * Plot results and save * //
177  RooPlot *frame = invMassVar.frame(); // create a frame to draw
178  frame->SetAxisRange(particle.PlotFrom(), particle.PlotUntil());
179  invMassDistribution.plotOn(frame, // draw distribution
180  RooFit::LineWidth(2), RooFit::LineColor(kBlue+2), RooFit::LineWidth(1),
181  RooFit::MarkerColor(kBlue+2), RooFit::MarkerSize(.5));
182  doublegauss.plotOn(frame, RooFit::LineWidth(2), RooFit::LineColor(kBlack)); // draw sig+bck fit
183  doublegauss.plotOn(frame, RooFit::Components(gauss1), // draw gauss 1
184  RooFit::LineWidth(1), RooFit::LineColor(kRed-4));
185  doublegauss.plotOn(frame, RooFit::Components(gauss2), // draw gauss 2
186  RooFit::LineWidth(1), RooFit::LineColor(kBlue-4));
187  doublegauss.paramOn(frame, RooFit::Layout(.56, .98, .92));
188 
189  // * Write fitted histograms * //
190  TCanvas c;
191  c.SetBatch();
192  frame->Draw();
193  const TString outputDir = Form("%s/%s", Settings::Output::PlotOutputDir.Data(), __BASE_FILE__);
194  gSystem->mkdir(outputDir.Data());
195  c.SaveAs(Form("%s/DoubleGauss_%s.%s", outputDir.Data(), particle.GetName(), Settings::Output::Extension));
196  c.Close();
