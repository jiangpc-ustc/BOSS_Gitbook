468  void rhopi_pipigg::ComputeInvariantMasses(KalmanKinematicFit *kkmfit)
469  {
470  // * Get vectors * //
471  HepLorentzVector ppim = kkmfit->pfit(0);
472  HepLorentzVector ppip = kkmfit->pfit(1);
473  HepLorentzVector pgamma1 = kkmfit->pfit(2);
474  HepLorentzVector pgamma2 = kkmfit->pfit(3);
475  // * Primary candidates * //
476  HepLorentzVector ppi0 = pgamma1 + pgamma2;
477  HepLorentzVector prho0 = ppim + ppip;
478  HepLorentzVector prhom = ppi0 + ppim;
479  HepLorentzVector prhop = ppi0 + ppip;
480  // * Three J/psi candidates * //
481  HepLorentzVector pJpsiRho0 = prho0 + ppi0;
482  HepLorentzVector pJpsiRhom = prhom + ppip;
483  HepLorentzVector pJpsiRhop = prhop + ppim;
484  // * Compute gamma energies and angle between the two photons * //
485  fE_gamma1 = pgamma1.e();
486  fE_gamma2 = pgamma2.e();
487  fCosGamma = abs(fE_gamma1-fE_gamma2) / ppi0.rho();
488  // * Compute invariant masses * //
489  fM_pi0 = ppi0.m();
490  fM_rho0 = prho0.m();
491  fM_rhom = prhom.m();
492  fM_rhop = prhop.m();
493  fM_JpsiRho0 = pJpsiRho0.m();
494  fM_JpsiRhom = pJpsiRhom.m();
495  fM_JpsiRhop = pJpsiRhop.m();
496  }
