473  void rhopi_pipigg::ComputeInvariantMasses(KalmanKinematicFit *kkmfit)
474  {
475  // * Get vectors * //
476  HepLorentzVector ppim = kkmfit->pfit(0);
477  HepLorentzVector ppip = kkmfit->pfit(1);
478  HepLorentzVector pgamma1 = kkmfit->pfit(2);
479  HepLorentzVector pgamma2 = kkmfit->pfit(3);
480  // * Primary candidates * //
481  HepLorentzVector ppi0 = pgamma1 + pgamma2;
482  HepLorentzVector prho0 = ppim + ppip;
483  HepLorentzVector prhom = ppi0 + ppim;
484  HepLorentzVector prhop = ppi0 + ppip;
485  // * Three J/psi candidates * //
486  HepLorentzVector pJpsiRho0 = prho0 + ppi0;
487  HepLorentzVector pJpsiRhom = prhom + ppip;
488  HepLorentzVector pJpsiRhop = prhop + ppim;
489  // * Compute gamma energies and angle between the two photons * //
490  fE_gamma1 = pgamma1.e();
491  fE_gamma2 = pgamma2.e();
492  fCosGamma = abs(fE_gamma1-fE_gamma2) / ppi0.rho();
493  // * Compute invariant masses * //
494  fM_pi0 = ppi0.m();
495  fM_rho0 = prho0.m();
496  fM_rhom = prhom.m();
497  fM_rhop = prhop.m();
498  fM_JpsiRho0 = pJpsiRho0.m();
499  fM_JpsiRhom = pJpsiRhom.m();
500  fM_JpsiRhop = pJpsiRhop.m();
501  }

507  void rhopi_pipigg::SetSmallestAngles(std::vector<EvtRecTrack*>::iterator &iter, std::vector<EvtRecTrack*> &vec, Hep3Vector &emcpos)
508  {
509  for(iter = vec.begin(); iter != vec.end(); ++iter) {
510  // * Get the extension object from MDC to EMC
511  if(!(*iter)->isExtTrackValid()) continue;
512  RecExtTrack *extTrk = (*iter)->extTrack();
513  if(extTrk->emcVolumeNumber() == -1) continue;
514  Hep3Vector extpos(extTrk->emcPosition());
515 
516  // * Get angles
517  // double cosTheta = extpos.cosTheta(emcpos);
518  double angle = extpos.angle(emcpos);
519  double dTheta = extpos.theta() - emcpos.theta();
520  double dPhi = extpos.deltaPhi(emcpos);
521  dTheta = fmod(dTheta + CLHEP::twopi + CLHEP::twopi + pi, CLHEP::twopi) - CLHEP::pi;
522  dPhi = fmod(dPhi + CLHEP::twopi + CLHEP::twopi + pi, CLHEP::twopi) - CLHEP::pi;
523  if(angle < fSmallestAngle){
524  fSmallestAngle = angle;
525  fSmallestTheta = dTheta;
526  fSmallestPhi = dPhi;
527  }
528  }
