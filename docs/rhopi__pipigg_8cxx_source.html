469  void rhopi_pipigg::ComputeInvariantMasses(KalmanKinematicFit *kkmfit)
470  {
471  // * Get vectors * //
472  HepLorentzVector ppim = kkmfit->pfit(0);
473  HepLorentzVector ppip = kkmfit->pfit(1);
474  HepLorentzVector pgamma1 = kkmfit->pfit(2);
475  HepLorentzVector pgamma2 = kkmfit->pfit(3);
476  // * Primary candidates * //
477  HepLorentzVector ppi0 = pgamma1 + pgamma2;
478  HepLorentzVector prho0 = ppim + ppip;
479  HepLorentzVector prhom = ppi0 + ppim;
480  HepLorentzVector prhop = ppi0 + ppip;
481  // * Three J/psi candidates * //
482  HepLorentzVector pJpsiRho0 = prho0 + ppi0;
483  HepLorentzVector pJpsiRhom = prhom + ppip;
484  HepLorentzVector pJpsiRhop = prhop + ppim;
485  // * Compute gamma energies and angle between the two photons * //
486  fE_gamma1 = pgamma1.e();
487  fE_gamma2 = pgamma2.e();
488  fCosGamma = abs(fE_gamma1-fE_gamma2) / ppi0.rho();
489  // * Compute invariant masses * //
490  fM_pi0 = ppi0.m();
491  fM_rho0 = prho0.m();
492  fM_rhom = prhom.m();
493  fM_rhop = prhop.m();
494  fM_JpsiRho0 = pJpsiRho0.m();
495  fM_JpsiRhom = pJpsiRhom.m();
496  fM_JpsiRhop = pJpsiRhop.m();
497  }

503  void rhopi_pipigg::SetSmallestAngles(std::vector<EvtRecTrack*>::iterator &iter, std::vector<EvtRecTrack*> &vec, Hep3Vector &emcpos)
504  {
505  for(iter = vec.begin(); iter != vec.end(); ++iter) {
506  // * Get the extension object from MDC to EMC
507  if(!(*iter)->isExtTrackValid()) continue;
508  RecExtTrack *extTrk = (*iter)->extTrack();
509  if(extTrk->emcVolumeNumber() == -1) continue;
510  Hep3Vector extpos(extTrk->emcPosition());
511 
512  // * Get angles
513  // double cosTheta = extpos.cosTheta(emcpos);
514  double angle = extpos.angle(emcpos);
515  double dTheta = extpos.theta() - emcpos.theta();
516  double dPhi = extpos.deltaPhi(emcpos);
517  dTheta = fmod(dTheta + CLHEP::twopi + CLHEP::twopi + pi, CLHEP::twopi) - CLHEP::pi;
518  dPhi = fmod(dPhi + CLHEP::twopi + CLHEP::twopi + pi, CLHEP::twopi) - CLHEP::pi;
519  if(angle < fSmallestAngle){
520  fSmallestAngle = angle;
521  fSmallestTheta = dTheta;
522  fSmallestPhi = dPhi;
523  }
524  }
