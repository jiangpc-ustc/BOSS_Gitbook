200  fEventHeader = SmartDataPtr<Event::EventHeader>(eventSvc(), "/Event/EventHeader");
201  fEvtRecEvent = SmartDataPtr<EvtRecEvent> (eventSvc(), EventModel::EvtRec::EvtRecEvent);
202  fEvtRecTrkCol = SmartDataPtr<EvtRecTrackCol> (eventSvc(), EventModel::EvtRec::EvtRecTrackCol);

209  << "Ncharged = " << fEvtRecEvent->totalCharged() << ", "
210  << "Nneutral = " << fEvtRecEvent->totalNeutral() << ", "
211  << "Ntotal = " << fEvtRecEvent->totalTracks() << endmsg;
212 
213  // * Set vertex origin *
214  IVertexDbSvc* vtxsvc;
215  Gaudi::svcLocator()->service("VertexDbSvc", vtxsvc);
216  double v0x, v0y, v0z;
217  if(vtxsvc->isVertexValid()){
218  double* dbv = vtxsvc->PrimaryVertex();
219  double* vv = vtxsvc->SigmaPrimaryVertex();
220  v0x = dbv[0];
221  v0y = dbv[1];
222  v0z = dbv[2];
223  }
224 
225 
227 
228  // * Print log and set counters *
229  fLog << MSG::DEBUG << "Starting 'good' charged track selection:" << endmsg;
230  int nChargesMDC = 0;
231  ParticleID *fPID = ParticleID::instance();
232 
233  // * Loop over charged tracks *
234  fGoodChargedTracks.clear();
235  for(int i = 0; i < fEvtRecEvent->totalCharged(); ++i) {
236  // Note: the first part of the set of reconstructed tracks are the charged tracks
239 
240  // * Get track info from Main Drift Chamber
241  fLog << MSG::DEBUG << " charged track " << i << "/" << fEvtRecEvent->totalCharged() << endmsg;
242  fTrackIterator = fEvtRecTrkCol->begin() + i;
243  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
244  fTrackMDC = (*fTrackIterator)->mdcTrack();
245 
246  // * Get kinematics of track
247  double phi = fTrackMDC->helix(1);
248  double vr =
249  (fTrackMDC->x() - v0x) * cos(phi) +
250  (fTrackMDC->y() - v0y) * sin(phi);
251 
252  // * Get radii of track vertex
253  HepVector a = fTrackMDC->helix();
254  HepSymMatrix Ea = fTrackMDC->err();
255  HepPoint3D point0(0., 0., 0.); // initial point for MDC reconstruction
256  HepPoint3D IP(v0x, v0y, v0z);
257  VFHelix helixip(point0, a, Ea);
258  helixip.pivot(IP);
259  HepVector vecipa = helixip.a();
260  double rvxy = fabs(vecipa[0]); // nearest distance to IP in xy plane
261  double rvz = vecipa[3]; // nearest distance to IP in z direction
262  double rvphi = vecipa[1]; // angle in the xy-plane (?)
263 
265 
266  // * Apply vertex cuts
267  if(fTrackMDC->z() >= fCut_vz0) continue;
268  if(vr >= fCut_vr0) continue;
269  if(rvz >= fCut_rvz0) continue;
270  if(rvxy >= fCut_rvxy0) continue;
271 
272  // * Add charged track to vector
273  fGoodChargedTracks.push_back(*fTrackIterator);
274  nChargesMDC += fTrackMDC->charge(); // @todo Check if this makes sense at all
275 
277  if(fDo_charged) {
278  fCharged.at("vx") = fTrackMDC->x();
279  fCharged.at("vy") = fTrackMDC->y();
280  fCharged.at("vz") = fTrackMDC->z();
281  fCharged.at("vr") = vr;
282  fCharged.at("rvxy") = rvxy;
283  fCharged.at("rvz") = rvz;
284  fCharged.at("rvphi") = rvphi;
285  fCharged.at("phi") = phi;
286  fCharged.at("p") = fTrackMDC->p();
287  fNTupleMap.at("charged")->write();
288  }
289 
291  if(fDo_dedx) WriteDedxInfo(*fTrackIterator, "dedx", fDedx);
292 
294  if(fDo_ToFEC || fDo_ToFIB || fDo_ToFOB) {
295 
296  // * Check if MDC and TOF info for track are valid *
297  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
298  if(!(*fTrackIterator)->isTofTrackValid()) continue;
299 
300  // * Get momentum as determined by MDC *
301  fTrackMDC = (*fTrackIterator)->mdcTrack();
302  double ptrk;
303  if(fTrackMDC) fTrackMDC->p();
304  SmartRefVector<RecTofTrack> tofTrkCol = (*fTrackIterator)->tofTrack();
305  SmartRefVector<RecTofTrack>::iterator iter_tof = tofTrkCol.begin();
306  for(; iter_tof != tofTrkCol.end(); ++iter_tof) {
307  TofHitStatus hitStatus;
308  hitStatus.setStatus((*iter_tof)->status());
309  if(!hitStatus.is_counter()) continue;
310  if(hitStatus.is_barrel()) {
311  if(hitStatus.layer() == 1) { // inner barrel
312  if(fDo_ToFIB) WriteTofInformation(iter_tof, ptrk, "ToFIB", fTofIB);
313  } else if(hitStatus.layer() == 2) { // outer barrel
314  if(fDo_ToFOB) WriteTofInformation(iter_tof, ptrk, "ToFOB", fTofOB);

316  }
317  else if(fDo_ToFEC && hitStatus.layer() == 0) // end cap
318  WriteTofInformation(iter_tof, ptrk, "ToFEC", fTofEC);

320 
321  } // if(fDo_tofec || fDo_tofib || fDo_tofob)
322 
324  }
325 
326 
329  fGoodNeutralTracks.clear();

333  fLog << MSG::DEBUG << " neutral track " << i-fEvtRecEvent->totalCharged() << "/" << fEvtRecEvent->totalNeutral() << endmsg;
334  fTrackIterator = fEvtRecTrkCol->begin() + i;
335  if(!(*fTrackIterator)->isEmcShowerValid()) continue;
336  fTrackEMC = (*fTrackIterator)->emcShower();
337  if(!fTrackEMC) continue;
338 
340  if(fTrackEMC->energy() < fMaxPhotonEnergy) continue;
341 
343  if(fDo_neutral) {
344  fNeutral.at("E") = fTrackEMC->energy();
345  fNeutral.at("x") = fTrackEMC->x();

358  }

411  {

413  double eraw = emcTrk->energy();
414  double phi = emcTrk->phi();
415  double the = emcTrk->theta();
416  HepLorentzVector ptrk(
417  eraw * sin(the) * cos(phi), // px
418  eraw * sin(the) * sin(phi), // py
419  eraw * cos(the), // pz
420  eraw);

495  template<typename TYPE>
496  void TrackSelector::WriteTofInformation(SmartRefVector<RecTofTrack>::iterator iter_tof, double ptrk, const char* tupleName, std::map<std::string, NTuple::Item<TYPE> > &map)
497  {
498 
499  // * Get ToF for each particle hypothesis
500  double path = (*iter_tof)->path();
501  std::vector<double> texp(gNMasses);
502  for(size_t j = 0; j < texp.size(); ++j) {
503  double gb = ptrk/gMasses[j]; // v = p/m (non-relativistic velocity)
504  double beta = gb/sqrt(1+gb*gb);
505  texp[j] = 10 * path /beta/gSpeedOfLight; // hypothesis ToF
506  }
507 
508  // * <b>write</b> ToF info
509  map.at("p") = ptrk;
510  map.at("tof") = (*iter_tof)->tof();
511  map.at("path") = (*iter_tof)->path();
512  map.at("cntr") = (*iter_tof)->tofID();
513  map.at("ph") = (*iter_tof)->ph();
514  map.at("zrhit") = (*iter_tof)->zrhit();
515  map.at("qual") = (*iter_tof)->quality();
516  map.at("te") = path - texp[0];
517  map.at("tmu") = path - texp[1];
518  map.at("tpi") = path - texp[2];
519  map.at("tk") = path - texp[3];
520  map.at("tp") = path - texp[4];
521  fNTupleMap.at(tupleName)->write();
522 
523  }
