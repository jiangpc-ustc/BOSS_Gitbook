202  {
203  fLog << MSG::DEBUG << "===>> TrackSelector::execute() <<===" << endmsg;
204 
212 
213  // * Load headers
214  fLog << MSG::DEBUG << "Loading EventHeader, EvtRecEvent, and EvtRecTrackCol" << endmsg;
215  fEventHeader = SmartDataPtr<Event::EventHeader>(eventSvc(), "/Event/EventHeader");
216  fEvtRecEvent = SmartDataPtr<EvtRecEvent> (eventSvc(), EventModel::EvtRec::EvtRecEvent);
217  fEvtRecTrkCol = SmartDataPtr<EvtRecTrackCol> (eventSvc(), EventModel::EvtRec::EvtRecTrackCol);
218 
219  // * Log run number, event number, and number of events *
220  fLog << MSG::DEBUG
221  << "RUN " << fEventHeader->runNumber() << ", "
222  << "event number " << fEventHeader->eventNumber() << endmsg;
223  fLog << MSG::DEBUG
224  << "Ncharged = " << fEvtRecEvent->totalCharged() << ", "
225  << "Nneutral = " << fEvtRecEvent->totalNeutral() << ", "
226  << "Ntotal = " << fEvtRecEvent->totalTracks() << endmsg;
227 
228  // * Increase counters *
229  ++fCounter_Nevents;
230  fCounter_Ncharged += fEvtRecEvent->totalCharged();
231  fCounter_Nneutral += fEvtRecEvent->totalNeutral();
232  fCounter_Ntracks += fEvtRecEvent->totalTracks();
233 
234  // * Set vertex origin *
235  IVertexDbSvc* vtxsvc;
236  Gaudi::svcLocator()->service("VertexDbSvc", vtxsvc);
237  if(vtxsvc->isVertexValid()){
238  double* dbv = vtxsvc->PrimaryVertex();
239  // double* vv = vtxsvc->SigmaPrimaryVertex();
240  fVertexPoint.set(dbv[0], dbv[1], dbv[2]);
241  }
242 
243 
246 
247  // * Reset counters and initialize PID instance *
248  fGoodChargedTracks.clear();
249  fPIDInstance = ParticleID::instance();
250 
251  // * Only perform if there are charged tracks *
252  if(fDo_charged && fEvtRecEvent->totalCharged()) {
253 
254  fNChargesMDC = 0;
255  fLog << MSG::DEBUG << "Starting 'good' charged track selection:" << endmsg;
256  for(int i = 0; i < fEvtRecEvent->totalCharged(); ++i) {
259 
260  // * Get track info from Main Drift Chamber
261  fLog << MSG::DEBUG << " charged track " << i << "/" << fEvtRecEvent->totalCharged() << endmsg;
262  fTrackIterator = fEvtRecTrkCol->begin() + i;
263  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
264  ++fCounter_Nmdcvalid;
265  fTrackMDC = (*fTrackIterator)->mdcTrack();
266 
267  // * Get kinematics of track
268  double phi = fTrackMDC->helix(1);
269  double vr =
270  (fTrackMDC->x() - fVertexPoint.x()) * cos(phi) +
271  (fTrackMDC->y() - fVertexPoint.y()) * sin(phi);
272 
273  // * Get radii of track vertex
274  HepVector a = fTrackMDC->helix();
275  HepSymMatrix Ea = fTrackMDC->err();
276  HepPoint3D point0(0., 0., 0.); // initial point for MDC reconstruction
277  VFHelix helixip(point0, a, Ea);
278  helixip.pivot(fVertexPoint);
279  HepVector vecipa = helixip.a();
280  double rvxy = fabs(vecipa[0]); // nearest distance to IP in xy plane
281  double rvz = vecipa[3]; // nearest distance to IP in z direction
282  double rvphi = vecipa[1]; // angle in the xy-plane (?)
283 
286  if(fCut_Vz.FailsMax(fabs(fTrackMDC->z()))) continue;
287  if(fCut_Vxy.FailsMax(vr) ) continue;
288  if(fCut_Rz .FailsMax(fabs(rvz)) ) continue;
289  if(fCut_Rxy.FailsMax(rvxy) ) continue;
290 
292  // * Add charged track to vector
293  fGoodChargedTracks.push_back(*fTrackIterator);
294  fNChargesMDC += fTrackMDC->charge(); // @todo Check if this makes sense at all
295 
297  if(fWrite_charged) {
298  fMap_charged.at("vx") = fTrackMDC->x();
299  fMap_charged.at("vy") = fTrackMDC->y();
300  fMap_charged.at("vz") = fTrackMDC->z();
301  fMap_charged.at("vr") = vr;
302  fMap_charged.at("rvxy") = rvxy;
303  fMap_charged.at("rvz") = rvz;
304  fMap_charged.at("rvphi") = rvphi;
305  fMap_charged.at("phi") = phi;
306  fMap_charged.at("p") = fTrackMDC->p();
307  fNTupleMap.at("charged")->write();
308  }
309 
311  if(fWrite_dedx) WriteDedxInfo(*fTrackIterator, "dedx", fMap_dedx);
312 
314  if(fWrite_ToFEC || fWrite_ToFIB || fWrite_ToFOB) {
315 
316  // * Check if MDC and TOF info for track are valid *
317  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
318  if(!(*fTrackIterator)->isTofTrackValid()) continue;
319 
320  // * Get momentum as determined by MDC *
321  fTrackMDC = (*fTrackIterator)->mdcTrack();
322  double ptrk;
323  if(fTrackMDC) ptrk = fTrackMDC->p();
324  SmartRefVector<RecTofTrack> tofTrkCol = (*fTrackIterator)->tofTrack();
325  SmartRefVector<RecTofTrack>::iterator iter_tof = tofTrkCol.begin();
326  for(; iter_tof != tofTrkCol.end(); ++iter_tof) {
327  TofHitStatus hitStatus;
328  hitStatus.setStatus((*iter_tof)->status());
329  if(!hitStatus.is_counter()) continue;
330  if(hitStatus.is_barrel()) {
331  if(hitStatus.layer() == 1) { // inner barrel
332  if(fWrite_ToFIB) WriteTofInformation(iter_tof, ptrk, "ToFIB", fMap_TofIB);
333  } else if(hitStatus.layer() == 2) { // outer barrel
334  if(fWrite_ToFOB) WriteTofInformation(iter_tof, ptrk, "ToFOB", fMap_TofOB);
335  }
336  }
337  else if(fWrite_ToFEC && hitStatus.layer() == 0) // end cap
338  WriteTofInformation(iter_tof, ptrk, "ToFEC", fMap_TofEC);
339  }
340 
341  } // if(fWrite_tofec || fWrite_tofib || fWrite_tofob)
342 
344  }
345 
346  // * Finish good charged track selection *
347  fLog << MSG::DEBUG << "Number of 'good' charged tracks: " << fGoodChargedTracks.size() << endmsg;
348  }
349 
350 
353 
354  // * Set counters and data members *
355  fGoodNeutralTracks.clear();
356 
357  // * Only perform if there are neutral tracks *
358  if(fDo_neutral && fEvtRecEvent->totalNeutral()) {
359 
360  // * Loop over neutral tracks *
361  fLog << MSG::DEBUG << "Starting 'good' neutral track selection:" << endmsg;
362  for(int i = fEvtRecEvent->totalCharged(); i < fEvtRecEvent->totalTracks(); ++i) {
365  fLog << MSG::DEBUG << " neutral track " << i-fEvtRecEvent->totalCharged() << "/" << fEvtRecEvent->totalNeutral() << endmsg;
366  fTrackIterator = fEvtRecTrkCol->begin() + i;
367  if(!(*fTrackIterator)->isEmcShowerValid()) continue;
368  fTrackEMC = (*fTrackIterator)->emcShower();
369  if(!fTrackEMC) continue;
370 
372  if(fCut_PhotonEnergy.FailsMin(fTrackEMC->energy())) continue;
373 
375  if(fWrite_neutral) {
376  fMap_neutral.at("E") = fTrackEMC->energy();
377  fMap_neutral.at("x") = fTrackEMC->x();
378  fMap_neutral.at("y") = fTrackEMC->y();
379  fMap_neutral.at("z") = fTrackEMC->z();
380  fMap_neutral.at("phi") = fTrackEMC->phi();
381  fMap_neutral.at("theta") = fTrackEMC->theta();
382  fMap_neutral.at("time") = fTrackEMC->time();
383  fNTupleMap.at("neutral")->write();
384  }
385 
387  fGoodNeutralTracks.push_back(*fTrackIterator);
388 
390  }
391 
392  // * Finish good photon selection *
393  fLog << MSG::DEBUG << "Number of 'good' photons: " << fGoodNeutralTracks.size() << endmsg;
394  }
395 
396 
398  if(fWrite_mult) {
399  fMap_mult.at("Ntotal") = fEvtRecEvent->totalTracks();
400  fMap_mult.at("Ncharge") = fEvtRecEvent->totalCharged();
401  fMap_mult.at("Nneutral") = fEvtRecEvent->totalNeutral();
402  if(fDo_charged) {
403  fMap_mult.at("NgoodCharged") = fGoodChargedTracks.size();
404  fMap_mult.at("Nmdc") = fNChargesMDC;
405  }
406  if(fDo_neutral) fMap_mult.at("NgoodNeutral") = fGoodNeutralTracks.size();
407  fNTupleMap.at("mult")->write();
408  }
409 
411  if(fWrite_vertex) {
412  fMap_vertex.at("vx0") = fVertexPoint.x();
413  fMap_vertex.at("vy0") = fVertexPoint.y();
414  fMap_vertex.at("vz0") = fVertexPoint.z();
415  fNTupleMap.at("vertex")->write();
416  }
417 
419  if(execute_rest() == StatusCode::SUCCESS) return StatusCode::SUCCESS;
420  if(fPIDInstance) delete fPIDInstance;
421 
423  return StatusCode::SUCCESS;
424  }

454  HepLorentzVector TrackSelector::ComputeMomentum(EvtRecTrack *track)
455  {
456  RecEmcShower *emcTrk = track->emcShower();
457  double eraw = emcTrk->energy();
458  double phi = emcTrk->phi();
459  double the = emcTrk->theta();
460  HepLorentzVector ptrk(
461  eraw * sin(the) * cos(phi), // px
462  eraw * sin(the) * sin(phi), // py
463  eraw * cos(the), // pz
464  eraw);
465  // ptrk = ptrk.boost(-0.011, 0, 0); // boost to center-of-mass frame
466  return ptrk;
467  }

729  void TrackSelector::WriteTofInformation(SmartRefVector<RecTofTrack>::iterator iter_tof, double ptrk, const char* tupleName, std::map<std::string, NTuple::Item<double> > &map)
730  {
731 
732  // * Get ToF for each particle hypothesis
733  double path = (*iter_tof)->path();
734  std::vector<double> texp(gNMasses);
735  for(size_t j = 0; j < texp.size(); ++j) {
736  double gb = ptrk/gMasses[j]; // v = p/m (non-relativistic velocity)
737  double beta = gb/sqrt(1+gb*gb);
738  texp[j] = 10 * path /beta/gSpeedOfLight; // hypothesis ToF
739  }
740 
741  // * <b>write</b> ToF info
742  map.at("p") = ptrk;
743  map.at("tof") = (*iter_tof)->tof();

747  map.at("zrhit") = (*iter_tof)->zrhit();
748  map.at("qual") = (*iter_tof)->quality();
749  map.at("tof_e") = path - texp[0];

751  map.at("tof_pi") = path - texp[2];
