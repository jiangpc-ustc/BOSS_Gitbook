360  void BOSSRootFile::LoadTrees(bool print)
361  {
362  // * Go through keys in the file and add to map of TTrees *
363  TIter next(fFile.GetListOfKeys());
364  TObject *obj = nullptr;
365  TKey *key = nullptr;
366  TTree *tree = nullptr;
367  while((obj = next())) {
368  if((key = dynamic_cast<TKey*>(obj))) {
369  if((tree = dynamic_cast<TTree*>(key->ReadObj()))) {
370  fTrees.emplace(tree->GetName(), SimplifiedTree(tree, print));
371  }
372  }
373  }
374  // * Output *
