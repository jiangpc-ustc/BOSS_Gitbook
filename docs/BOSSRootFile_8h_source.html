351  std::list<SimplifiedTree*> BOSSRootFile::CreateOrderedMap()
352  {
353  // * Create list of pointers to `SimplifiedTree` * //
354  std::list<SimplifiedTree*> outputList;
355  for(auto it = fTrees.begin(); it != fTrees.end(); ++it) outputList.push_back(&(it->second));
356  // * Sort resulting list based on number of entries * //
357  outputList.sort([](SimplifiedTree* const & a, SimplifiedTree* const & b)
358  {
359  std::string nameA = a->Get()->GetName();
360  std::string nameB = b->Get()->GetName();
361  return a->GetEntries() != b->GetEntries() ?
362  a->GetEntries() > b->GetEntries() :
363  nameA < nameB;
364  });
365  return outputList;
366  }
