88  TTree* BOSSRootFile::GetTree(const char* treeName)
89  {
90  TTree* tree = NULL;
91  // * Loop over list of TTrees *
92  std::list<TTree*>::iterator it = fTTrees.begin();
93  while(it != fTTrees.end()) {
94  TString name = (*it)->GetName();
95  if(name.EqualTo(treeName)) {
96  tree = *it;
97  break;
98  }
99  ++it;
100  }
101  // * Pass TTree pointer if the name matches
102  if(!tree) {
103  std::cout << "ERROR: TTree \"" << treeName << "\" does not exist in file \"" << fFile->GetName() << "\"" << std::endl;
104  }
105  return tree;
106  }

163  void BOSSRootFile::LoadTTrees()
164  {
165  // * Only perform if file has been loaded *
166  if(!fFile) return;
167 
168  // * Go through keys in the file and add to list of TTrees *
169  TIter next(fFile->GetListOfKeys());
170  TObject *obj = NULL;
171  TKey *key = NULL;
172  TTree *tree = NULL;
173  while((obj = next())) {
174  if((key = dynamic_cast<TKey*>(obj))) {
175  if((tree = dynamic_cast<TTree*>(key->ReadObj()))) {
176  // * Add to list of TTrees
177  fTTrees.push_back(tree);
178  }
179  }
180  }
181 
182  // * Error output *
183  std::cout << "Loaded " << fTTrees.size() << " TTrees from file \"" << fFile->GetName() << "\"" << std::endl;
184  }
