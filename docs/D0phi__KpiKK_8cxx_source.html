277  void D0phi_KpiKK::ComputeInvariantMasses(KalmanKinematicFit *kkmfit)
278  {
279  HepLorentzVector pD0 = kkmfit->pfit(0) + kkmfit->pfit(1); // K^- pi^+
280  HepLorentzVector pphi = kkmfit->pfit(2) + kkmfit->pfit(3); // K^- K^+
