321 void FitGaussian(TH1D* invMassHistogram, const Particle& particle, const double sigma)
322 {
323  // * Check if null pointer and create clone if successful
324  if(!invMassHistogram) {
325  std::cout << "ERROR: argument is a null pointer" << std::endl;
326  return;
327  }
328 
329  // * Set RooFit fit variable (in this case only the invariant mass axis) *
330  double hmin = invMassHistogram->GetXaxis()->GetXmin();
331  double hmax = invMassHistogram->GetXaxis()->GetXmax();
332  RooRealVar invMassVar("invMassVar", invMassHistogram->GetXaxis()->GetTitle(), hmin, hmax);
333 
334  // * Import histogram to Roofit *
335  RooDataHist invMassDistribution(invMassHistogram->GetName(), invMassHistogram->GetTitle(), invMassVar, Import(*invMassHistogram));
336 
337  // * Create Gaussian function *
338  RooRealVar sigmean ("sigmean", Form("%s mass", particle.GetNameLaTeX()),
339  particle.GetMass()*1.03, particle.GetMass()*.95, particle.GetMass()*1.05);
340  RooRealVar sigwidth("sigwidth", Form("%s width", particle.GetNameLaTeX()), sigma);
341  RooGaussian signal("gauss",
342  Form("Gaussian PDF for %s distribution", invMassHistogram->GetXaxis()->GetTitle()),
343  invMassVar, sigmean, sigwidth);
344 
345  // * Background function *
346  double boundary = 30000.;
347  RooRealVar p0("p0", "p0", 0., -10000., 10000.);
348  RooRealVar p1("p1", "p1", 0., -10000., 10000.);
349  RooRealVar p2("p2", "p2", 0., -10000., 10000.);
350  RooRealVar p3("p2", "p2", 0., -10000., 10000.);
351  RooPolynomial background("background",
352  "Background Probability Density Function (PDF)",
353  invMassVar, RooArgList(p0, p1, p2, p3));
354 
355  // * Fit background + gaussian to the distribution *
356  // See https://root.cern.ch/doc/master/classRooAddPdf.html
357  RooRealVar nsig("nsig", "number of signal events", 800., 0., boundary);
358  RooRealVar nbkg("nbkg", "number of background events", 0., 0., boundary);
359  RooAddPdf model("total_distribution", "sig+bck",
360  RooArgList(signal, background), RooArgList(nsig, nbkg));
361  model.fitTo(invMassDistribution);
362 
363  // * Plot results and save *
364  RooPlot *frame = invMassVar.frame(); // create a frame to draw
365  invMassDistribution.plotOn(frame); // draw distribution
366  model.plotOn(frame); // draw sig+bck fit
367  model.plotOn(frame, Components(signal), LineStyle(kDashed), LineColor(kRed)); // draw signal
368  model.plotOn(frame, Components(background), LineStyle(kDashed), LineColor(kGreen)); // draw background
369 
370  // * Save results *
371  TCanvas c;
372  frame->Draw();
373  c.SaveAs(Settings::Output::Extension+"/"+invMassHistogram->GetName()+"_fit."+Settings::Output::Extension);
374 }
