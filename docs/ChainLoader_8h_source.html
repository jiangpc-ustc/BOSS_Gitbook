117  void ChainLoader::BookAddresses(bool print)
118  {
119  if(!fChain.GetNbranches()) return;
121  TIter next(fChain.GetListOfBranches());
122  TObject *obj = nullptr;
123  if(print) {
124  std::cout << "Tree \"" << fChain.GetName() << "\" (" << fChain.GetEntries() << " entries)" << std::endl;
125  std::cout << fChain.GetTitle() << std::endl;
126  std::cout << " "
127  << std::setw(18) << std::left << "BRANCH NAME"
128  << std::setw(12) << std::right << "MEAN" << std::endl;
129  }
131  while((obj = next())) {
132  std::string type(obj->GetTitle());
133  if(print) std::cout << " "
134  << std::setw(18) << std::left << type
135  << std::setw(12) << std::right << ComputeMean(&fChain, obj->GetName()) << std::endl;
136  switch(type.back()) {
137  case 'B' : SetAddress(obj, fMap_B); break;
138  case 'b' : SetAddress(obj, fMap_b); break;
139  case 'S' : SetAddress(obj, fMap_S); break;
140  case 's' : SetAddress(obj, fMap_s); break;
141  case 'I' : SetAddress(obj, fMap_I); break;
142  case 'i' : SetAddress(obj, fMap_i); break;
143  case 'F' : SetAddress(obj, fMap_F); break;
144  case 'D' : SetAddress(obj, fMap_D); break;
145  case 'L' : SetAddress(obj, fMap_L); break;
146  case 'l' : SetAddress(obj, fMap_l); break;
147  case 'O' : SetAddress(obj, fMap_O); break;
148  default :
149  std::cout << "ERROR: Unable to book address for branch \"" << type << "\"" << std::endl;
150  }
151  }
152  if(print) std::cout << std::endl;
153  }
