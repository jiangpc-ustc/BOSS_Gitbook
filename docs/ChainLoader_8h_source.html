112  void ChainLoader::BookAddresses(bool print)
113  {
114  if(!fChain.GetNbranches()) return;
116  TIter next(fChain.GetListOfBranches());
117  TObject *obj = nullptr;
118  if(print) {
119  std::cout << "Tree \"" << fChain.GetName() << "\" (" << fChain.GetEntries() << " entries)" << std::endl;
120  std::cout << fChain.GetTitle() << std::endl;
121  std::cout << " "
122  << std::setw(18) << std::left << "BRANCH NAME"
123  << std::setw(12) << std::right << "MEAN" << std::endl;
124  }
126  while((obj = next())) {
127  std::string type(obj->GetTitle());
128  if(print) std::cout << " "
129  << std::setw(18) << std::left << type
130  << std::setw(12) << std::right << ComputeMean(&fChain, obj->GetName()) << std::endl;
131  switch(type.back()) {
132  case 'B' : SetAddress(obj, fMap_B); break;
133  case 'b' : SetAddress(obj, fMap_b); break;
134  case 'S' : SetAddress(obj, fMap_S); break;
135  case 's' : SetAddress(obj, fMap_s); break;
136  case 'I' : SetAddress(obj, fMap_I); break;
137  case 'i' : SetAddress(obj, fMap_i); break;
138  case 'F' : SetAddress(obj, fMap_F); break;
139  case 'D' : SetAddress(obj, fMap_D); break;
140  case 'L' : SetAddress(obj, fMap_L); break;
141  case 'l' : SetAddress(obj, fMap_l); break;
142  case 'O' : SetAddress(obj, fMap_O); break;
143  default :
144  std::cout << "ERROR: Unable to book address for branch \"" << type << "\"" << std::endl;
145  }
146  }
147  if(print) std::cout << std::endl;
148  }
