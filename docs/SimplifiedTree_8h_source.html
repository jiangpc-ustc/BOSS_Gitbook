107  SimplifiedTree::SimplifiedTree(TTree* fTree, bool print) : fTree(fTree)
108  {
109  if(!fTree) return;
110  TIter next(fTree->GetListOfBranches());
111  TObject *obj = nullptr;
112  if(print) {
113  std::cout << "Tree \"" << fTree->GetName() << "\" (" << fTree->GetEntries() << " entries)" << std::endl;
114  std::cout << fTree->GetTitle() << std::endl;
115  std::cout << " "
116  << std::setw(18) << std::left << "BRANCH NAME"
117  << std::setw(12) << std::right << "MEAN" << std::endl;
118  }
119  while((obj = next())) {
120  std::string type(obj->GetTitle());
121  if(print) std::cout << " "
122  << std::setw(18) << std::left << type
123  << std::setw(12) << std::right << ComputeMean(fTree, obj->GetName()) << std::endl;
124  switch(type.back()) {
125  case 'B' : SetAddress(obj, fMap_B); break;
126  case 'b' : SetAddress(obj, fMap_b); break;
127  case 'S' : SetAddress(obj, fMap_S); break;
128  case 's' : SetAddress(obj, fMap_s); break;
129  case 'I' : SetAddress(obj, fMap_I); break;
130  case 'i' : SetAddress(obj, fMap_i); break;
131  case 'F' : SetAddress(obj, fMap_F); break;
132  case 'D' : SetAddress(obj, fMap_D); break;
133  case 'L' : SetAddress(obj, fMap_L); break;
134  case 'l' : SetAddress(obj, fMap_l); break;
135  case 'O' : SetAddress(obj, fMap_O); break;
136  default :
137  std::cout << "ERROR: Unable to book address for branch \"" << type << "\"" << std::endl;
138  }
139  }
140  if(print) std::cout << std::endl;
141  }
