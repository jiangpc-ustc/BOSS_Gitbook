92  SimplifiedTree::SimplifiedTree(TTree* fTree, bool print) : fTree(fTree)
93  {
94  if(!fTree) return;
95  TIter next(fTree->GetListOfBranches());
96  TObject *obj = nullptr;
97  if(print) std::cout << " Tree \"" << fTree->GetName() << "\" (" << fTree->GetEntries() << " entries) has branches:" << std::endl;
98  while((obj = next())) {
99  std::string type(obj->GetTitle());
100  if(print) std::cout << " " << type << std::endl;
101  switch(type.back()) {
102  case 'B' : SetAddress(obj, fMap_B); break;
103  case 'b' : SetAddress(obj, fMap_b); break;
104  case 'S' : SetAddress(obj, fMap_S); break;
105  case 's' : SetAddress(obj, fMap_s); break;
106  case 'I' : SetAddress(obj, fMap_I); break;
107  case 'i' : SetAddress(obj, fMap_i); break;
108  case 'F' : SetAddress(obj, fMap_F); break;
109  case 'D' : SetAddress(obj, fMap_D); break;
110  case 'L' : SetAddress(obj, fMap_L); break;
111  case 'l' : SetAddress(obj, fMap_l); break;
112  case 'O' : SetAddress(obj, fMap_O); break;
113  default :
114  std::cout << "ERROR: Unable to book address for branch \"" << type << "\"" << std::endl;
115  }
116  }
117  }
