94  SimplifiedTree::SimplifiedTree(TTree* fTree, bool print) : fTree(fTree)
95  {
96  if(!fTree) return;
97  TIter next(fTree->GetListOfBranches());
98  TObject *obj = nullptr;
99  if(print) {
100  std::cout << "Tree \"" << fTree->GetName() << "\" (" << fTree->GetEntries() << " entries)" << std::endl;
101  std::cout << fTree->GetTitle() << std::endl;
102  std::cout << " "
103  << std::setw(18) << std::left << "BRANCH NAME"
104  << std::setw(12) << std::right << "MEAN" << std::endl;
105  }
106  while((obj = next())) {
107  std::string type(obj->GetTitle());
108  if(print) std::cout << " "
109  << std::setw(18) << std::left << type
110  << std::setw(12) << std::right << ComputeMean(fTree, obj->GetName()) << std::endl;
111  switch(type.back()) {
112  case 'B' : SetAddress(obj, fMap_B); break;
113  case 'b' : SetAddress(obj, fMap_b); break;
114  case 'S' : SetAddress(obj, fMap_S); break;
115  case 's' : SetAddress(obj, fMap_s); break;
116  case 'I' : SetAddress(obj, fMap_I); break;
117  case 'i' : SetAddress(obj, fMap_i); break;
118  case 'F' : SetAddress(obj, fMap_F); break;
119  case 'D' : SetAddress(obj, fMap_D); break;
120  case 'L' : SetAddress(obj, fMap_L); break;
121  case 'l' : SetAddress(obj, fMap_l); break;
122  case 'O' : SetAddress(obj, fMap_O); break;
123  default :
124  std::cout << "ERROR: Unable to book address for branch \"" << type << "\"" << std::endl;
125  }
126  }
127  if(print) std::cout << std::endl;
