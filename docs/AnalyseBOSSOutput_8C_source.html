233  int LoadConfiguration(const char* filename)
234  {
236  ifstream file(filename);
237  if(!file.is_open()) {
238  cout << "WARNING: Could not load configuration file \"" << filename << "\"" << endl;
239  return -1;
240  }
242  cout << endl << "LOADING CONFIGURATION FROM \"" << filename << "\"" << endl;
244  string line;
245  while(getline(file, line)) {
248  if(line.back()<' ') line.pop_back();
250  if(line.find('=') == string::npos && line.find(' ') == string::npos) continue;
252  RemoveLeading(line);
253  RemoveLeading(line, '\t');
255  string beginning = line.substr(0,2);
256  if(!beginning.compare("//")) continue; // C++ line comment
257  if(!beginning.compare("/*")) continue; // C++ begin block comment
258  if(beginning.front() == '*') continue; // C++ continue block comment
259  if(beginning.front() == '#') continue; // bash comment
260  if(beginning.front() == '%') continue; // LaTeX comment
262  string parname { GetParameterName (line) };
263  string parvalue { GetParameterValue(line) };
265  ArgPair_base::SetParameters(parname, parvalue);
267  }
269  ArgPair_base::PrintAll();
271  }

274  string GetParameterName(string line)
275  {
277  if(line.find('=') != string::npos) line.resize(line.find_first_of('='));
279  else if(line.find(' ') != string::npos) line.resize(line.find_first_of(' '));
281  RemoveTrailing(line);
282  RemoveTrailing(line, '\t');
284  return line;
285  }

288  string GetParameterValue(string line)
289  {
291  if(line.find('=') != string::npos) line = line.substr(line.find_first_of('=')+1);
293  else if(line.find(' ') != string::npos) line = line.substr(line.find_first_of(' ')+1);
295  RemoveLeading(line);
296  RemoveLeading(line, '\t');
298  if(!line.compare("\"\"")) line = ""; // if empty string
299  else Trim(line, 34); // 34 is ASCII for double quoation mark "
301  return line;
302  }
