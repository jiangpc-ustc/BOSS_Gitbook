188  void FitObject::BuildDoubleGaussian()
189  {
190  fMean = std_fix::make_unique<RooRealVar>(
191  Form("m_{%s}", fParticle.GetNameLaTeX()),
192  Form("%s mass", fParticle.GetNameLaTeX()),
193  fParticle.GetMass(), fParticle.GetLowerMass(), fParticle.GetUpperMass());
194  fSigma1 = std_fix::make_unique<RooRealVar>("#sigma_{1}",
195  Form("%s width 1", fParticle.GetNameLaTeX()),
196  fParticle.GetGaussianSmallWidth(),
197  Settings::Fit::fSigmaScaleFactorLow * fParticle.GetGaussianSmallWidth(),
198  Settings::Fit::fSigmaScaleFactorUp * fParticle.GetGaussianSmallWidth());
199  fSigma2 = std_fix::make_unique<RooRealVar>("#sigma_{2}",
200  Form("%s width 2", fParticle.GetNameLaTeX()),
201  fParticle.GetGaussianWideWidth(),
202  Settings::Fit::fSigmaScaleFactorLow * fParticle.GetGaussianWideWidth(),
203  Settings::Fit::fSigmaScaleFactorUp * fParticle.GetGaussianWideWidth());
204  fGaussian1 = std_fix::make_unique<RooGaussian>("gauss1",
205  Form("Gaussian PDF 1 for #it{M}_{%s} distribution", fParticle.GetDaughterLabel()),
206  *fRooRealVar, *fMean, *fSigma1);
207  fGaussian2 = std_fix::make_unique<RooGaussian>("gauss2",
208  Form("Gaussian PDF 2 for #it{M}_{%s} distribution", fParticle.GetDaughterLabel()),
209  *fRooRealVar, *fMean, *fSigma2);
210  fNGauss1 = std_fix::make_unique<RooRealVar>("N_{gaus1}", "N_{gaus1}", 1e2, 0., 1e6);
211  fNGauss2 = std_fix::make_unique<RooRealVar>("N_{gaus2}", "N_{gaus2}", 1e4, 0., 1e6);
212  fComponents.add(*fGaussian1);
213  fComponents.add(*fGaussian2);
214  fNContributions.add(*fNGauss1);
215  fNContributions.add(*fNGauss2);
216  }
