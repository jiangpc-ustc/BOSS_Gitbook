233  StatusCode DzeroPhi::execute()
234  {
235  // * STEP (A): Load next event from DST file * //
236 
237  // * Load event info *
238  /*
239  http://bes3.to.infn.it/Boss/7.0.2/html/namespaceEventModel_1_1EvtRec.html (namespace)
240  http://bes3.to.infn.it/Boss/7.0.2/html/classEvtRecEvent.html (class)
241  http://bes3.to.infn.it/Boss/7.0.2/html/EvtRecTrack_8h.html (typedef EvtRecTrackCol)
242  */
243  fLog << MSG::DEBUG << "Loading EventHeader, EvtRecEvent, and EvtRecTrackCol" << endreq;
244  fEventHeader = SmartDataPtr<Event::EventHeader>(eventSvc(), "/Event/EventHeader");
245  fEvtRecEvent = SmartDataPtr<EvtRecEvent> (eventSvc(), EventModel::EvtRec::EvtRecEvent);
246  fEvtRecTrkCol = SmartDataPtr<EvtRecTrackCol> (eventSvc(), EventModel::EvtRec::EvtRecTrackCol);
247 
248  // * Log run number, event number, and number of events *
249  fLog << MSG::DEBUG
250  << "RUN " << fEventHeader->runNumber() << ", "
251  << "event number " << fEventHeader->eventNumber() << endreq;
252  fLog << MSG::DEBUG
253  << "Ncharged = " << fEvtRecEvent->totalCharged() << ", "
254  << "Nneutral = " << fEvtRecEvent->totalNeutral() << ", "
255  << "Ntotal = " << fEvtRecEvent->totalTracks() << endreq;
256 
257  // * Set vertex origin *
258  IVertexDbSvc* vtxsvc;
259  Gaudi::svcLocator()->service("VertexDbSvc", vtxsvc);
260  double v0x, v0y, v0z;
261  if(vtxsvc->isVertexValid()){
262  double* dbv = vtxsvc->PrimaryVertex();
263  double* vv = vtxsvc->SigmaPrimaryVertex();
264  v0x = dbv[0];
265  v0y = dbv[1];
266  v0z = dbv[2];
267  }
268 
269 
270  // * STEP (B): Create selection charged tracks * //
271 
272  // * Print log and set counters *
273  fLog << MSG::DEBUG << "Starting 'good' charged track selection:" << endreq;
274  int nChargesMDC = 0;
275  ParticleID *pid = ParticleID::instance();
276 
277  // * Clear vectors of selected particles *
278  fGoodChargedTracks.clear();
279  fKaonNeg.clear();
280  fKaonPos.clear();
281  fPionPos.clear();
282 
283  // * Loop over charged tracks *
284  for(int i = 0; i < fEvtRecEvent->totalCharged(); ++i) {
285  // Note: the first part of the set of reconstructed tracks are the charged tracks
286 
287  // * STEP 1: Get MDC information
288 
289  // * Get track info from Main Drift Chamber
290  fLog << MSG::DEBUG << " charged track " << i << "/" << fEvtRecEvent->totalCharged() << endreq;
291  fTrackIterator = fEvtRecTrkCol->begin() + i;
292  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
293  RecMdcTrack *mdcTrk = (*fTrackIterator)->mdcTrack();
294 
295  // * Get kinematics of track
296  double phi = mdcTrk->helix(1);
297  double vr =
298  (mdcTrk->x() - v0x) * cos(phi) +
299  (mdcTrk->y() - v0y) * sin(phi);
300 
301  // * Get radii of track vertex
302  HepVector a = mdcTrk->helix();
303  HepSymMatrix Ea = mdcTrk->err();
304  HepPoint3D point0(0., 0., 0.); // initial point for MDC reconstruction
305  HepPoint3D IP(v0x, v0y, v0z);
306  VFHelix helixip(point0, a, Ea);
307  helixip.pivot(IP);
308  HepVector vecipa = helixip.a();
309  double rvxy = fabs(vecipa[0]); // nearest distance to IP in xy plane
310  double rvz = vecipa[3]; // nearest distance to IP in z direction
311  double rvphi = vecipa[1]; // angle in the xy-plane (?)
312 
313  // * STEP 2: Apply vertex cuts, store track, and write info
314 
315  // * Apply vertex cuts
316  if(mdcTrk->z() >= fVz0cut) continue;
317  if(vr >= fVr0cut) continue;
318  if(rvz >= fRvz0cut) continue;
319  if(rvxy >= fRvxy0cut) continue;
320 
321  // * Add charged track to vector
322  fGoodChargedTracks.push_back(*fTrackIterator);
323  nChargesMDC += mdcTrk->charge(); // @todo Check if this makes sense at all
324 
325  // * WRITE primary vertex position info ("vxyz" branch)
326  if(fDoTrackVertex) {
327  fTrackVertex["vx"] = mdcTrk->x();
328  fTrackVertex["vy"] = mdcTrk->y();
329  fTrackVertex["vz"] = mdcTrk->z();
330  fTrackVertex["vr"] = vr;
331  fTrackVertex["rvxy"] = rvxy;
332  fTrackVertex["rvz"] = rvz;
333  fTrackVertex["rvphi"] = rvphi;
334  fTrackVertex["phi"] = phi;
335  fTrackVertex["p"] = mdcTrk->p();
336  fNTupleMap["vxyz"]->write();
337  }
338 
339  // * STEP 3: Distinguish and store types of particles
340 
341  // * Initialise PID sub-system
342  // if(pid) delete pid;
343  pid->init();
344  pid->setMethod(pid->methodProbability());
345  // pid->setMethod(pid->methodLikelihood()); // for likelihood Method
346  pid->setChiMinCut(4);
347  pid->setRecTrack(*fTrackIterator);
348  pid->usePidSys(pid->useDedx() | pid->useTof1() | pid->useTof2() | pid->useTofE()); // use PID sub-system
349  pid->identify(pid->onlyPion() | pid->onlyKaon()); // seperater Pion/Kaon
350  pid->calculate();
351  if(!(pid->IsPidInfoValid())) continue;
352 
353  // * WRITE particle identification info ("pid" branch)
354  if(fDoPID) {
355  fPID["ptrk"] = mdcTrk->p();
356  fPID["cost"] = cos(mdcTrk->theta());
357  fPID["dedx"] = pid->chiDedx(2);
358  fPID["tof1"] = pid->chiTof1(2);
359  fPID["tof2"] = pid->chiTof2(2);
360  fPID["prob"] = pid->probPion();
361  fNTupleMap["pid"]->write();
362  }
363 
364  // * Identify type of charged particle and add to related vector (this package: kaon and pion) *
365  RecMdcKalTrack* mdcKalTrk = (*fTrackIterator)->mdcKalTrack(); // after ParticleID, use RecMdcKalTrack substitute RecMdcTrack
366  if(pid->probPion() > pid->probKaon()) { // NOTE: if more likely to be pion
367  if(pid->probPion() < 0.001) continue; // check if indeed a pion
368  if(pid->pdf(2) < pid->pdf(3)) continue; // for likelihood method (0=electron 1=muon 2=pion 3=kaon 4=proton)
369  RecMdcKalTrack::setPidType(RecMdcKalTrack::pion); // PID can be set to electron, muon, pion (default), kaon and proton
370  if(mdcKalTrk->charge() > 0) fPionPos.push_back(*fTrackIterator); // if positive pion
371  }
372  else { // NOTE: if more likely to be kaon (K+-)
373  if(pid->probKaon() < 0.001) continue; // check if indeed a kaon
374  if(pid->pdf(3) < pid->pdf(2)) continue; // for likelihood method (0=electron 1=muon 2=pion 3=kaon 4=proton)
375  RecMdcKalTrack::setPidType(RecMdcKalTrack::kaon); // PID can be set to electron, muon, pion (default), kaon and proton
376  if(mdcKalTrk->charge() < 0) fKaonNeg.push_back(*fTrackIterator); // if negative kaon
377  else fKaonPos.push_back(*fTrackIterator); // if positive kaon
378  }
379 
380  }
381 
382 
383  // * STEP (C): WRITE event info ("mult" and "vertex" branch) * //
384  fLog << MSG::DEBUG << "ngood, totcharge = " << fGoodChargedTracks.size() << " , " << nChargesMDC << endreq;
385  if(fDoMult) {
386  fMult["Ntotal"] = fEvtRecEvent->totalTracks();
387  fMult["Ncharge"] = fEvtRecEvent->totalCharged();
388  fMult["Nneutral"] = fEvtRecEvent->totalNeutral();
389  fMult["Ngood"] = fGoodChargedTracks.size();
390  fMult["Nmdc"] = nChargesMDC;
391  fMult["NKaonNeg"] = fKaonNeg.size();
392  fMult["NKaonPos"] = fKaonPos.size();
393  fMult["NPionPos"] = fPionPos.size();
394  fNTupleMap["vertex"]->write();
395  }
396  if(fDoVertex) {
397  fVertex["vx0"] = v0x;
398  fVertex["vy0"] = v0y;
399  fVertex["vz0"] = v0z;
400  fNTupleMap["vertex"]->write();
401  }
402 
403 
404  // * STEP (D): WRITE dE/dx PID information ("dedx" branch) * //
405  if(fDoDedx) WriteDedxInfoForVector(fGoodChargedTracks, "dedx", fDedx);
406  if(fDoDedx_K) WriteDedxInfoForVector(fGoodChargedTracks, "dedx_K", fDedx_K);
407  if(fDoDedx_K) WriteDedxInfoForVector(fGoodChargedTracks, "dedx_K", fDedx_K);
408  if(fDoDedx_pi) WriteDedxInfoForVector(fGoodChargedTracks, "dedx_pi", fDedx_pi);
409 
410 
411  // * STEP (E): WRITE Time-of-Flight PID information ("tof*" branch) * //
412  if(fDoTofEC || fDoTofIB || fDoTofOB) for(fTrackIterator = fGoodChargedTracks.begin(); fTrackIterator != fGoodChargedTracks.end(); ++fTrackIterator) {
413 
414  // * Check if MDC and TOF info for track are valid *
415  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
416  if(!(*fTrackIterator)->isTofTrackValid()) continue;
417 
418  // * Get momentum as determined by MDC *
419  RecMdcTrack * mdcTrk = (*fTrackIterator)->mdcTrack();
420  double ptrk = mdcTrk->p();
421  SmartRefVector<RecTofTrack> tofTrkCol = (*fTrackIterator)->tofTrack();
422  SmartRefVector<RecTofTrack>::iterator iter_tof = tofTrkCol.begin();
423  for(; iter_tof != tofTrkCol.end(); ++iter_tof) {
424  TofHitStatus hitStatus;
425  hitStatus.setStatus((*iter_tof)->status());
426  if(!hitStatus.is_counter()) continue;
427  if(hitStatus.is_barrel()) {
428  if(hitStatus.layer() == 1) { // inner barrel
429  if(fDoTofIB) WriteTofInformation(iter_tof, ptrk, "tof1", fTofIB);
430  } else if(hitStatus.layer() == 2) { // outer barrel
431  if(fDoTofOB) WriteTofInformation(iter_tof, ptrk, "tof2", fTofOB);
432  }
433  }
434  else if(fDoTofEC && hitStatus.layer() == 0) // end cap
435  WriteTofInformation(iter_tof, ptrk, "tofe", fTofEC);
436  }
437 
438  } // loop all charged tracks
439 
440 
441  // * STEP (F): WRITE Kalman 4-constraint kinematic fit with smallest chi squared ("fit4c" branch) * //
442  if(fDoFit4c) {
443 
444  // * Loop over all combinations and get the one with the smallest chi square
445  fSmallestChiSq = 9999999.;
446  KalmanKinematicFit *bestKalmanFit = KalmanKinematicFit::instance();
447  for(fKaonNeg1Iter = fKaonNeg.begin(); fKaonNeg1Iter != fKaonNeg.end(); ++fKaonNeg1Iter) {
448  for(fKaonNeg2Iter = fKaonNeg1Iter+1; fKaonNeg2Iter != fKaonNeg.end(); ++fKaonNeg2Iter) {
449  for(fKaonPosIter = fKaonPos.begin(); fKaonPosIter != fKaonPos.end(); ++fKaonPosIter) {
450  for(fPionPosIter = fPionPos.begin(); fPionPosIter != fPionPos.end(); ++fPionPosIter) {
451 
452  // * Get Kalman tracks
453  RecMdcKalTrack *KmTrk1 = (*fKaonNeg1Iter)->mdcKalTrack();
454  RecMdcKalTrack *KmTrk2 = (*fKaonNeg2Iter)->mdcKalTrack();
455  RecMdcKalTrack *KpTrk = (*fKaonPosIter )->mdcKalTrack();
456  RecMdcKalTrack *pipTrk = (*fPionPosIter )->mdcKalTrack();
457 
458  // * Get W-tracks
459  WTrackParameter wvKmTrk1(mK, KmTrk1->getZHelix(), KmTrk1->getZError());
460  WTrackParameter wvKmTrk2(mK, KmTrk2->getZHelix(), KmTrk2->getZError());
461  WTrackParameter wvKpTrk (mK, KpTrk->getZHelix(), KpTrk->getZError());
462  WTrackParameter wvpipTrk(mpi, pipTrk->getZHelix(), pipTrk->getZError());
463 
464  // * Test vertex fit * //
465  HepPoint3D vx(0., 0., 0.);
466  HepSymMatrix Evx(3, 0);
467  double bx = 1E+6;
468  double by = 1E+6;
469  double bz = 1E+6;
470  Evx[0][0] = bx*bx;
471  Evx[1][1] = by*by;
472  Evx[2][2] = bz*bz;
473 
474  VertexParameter vxpar;
475  vxpar.setVx(vx);
476  vxpar.setEvx(Evx);
477 
478  VertexFit* vtxfit = VertexFit::instance();
479  vtxfit->init();
480  vtxfit->AddTrack(0, wvpipTrk);
481  vtxfit->AddTrack(1, wvKpTrk);
482  vtxfit->AddTrack(2, wvKmTrk1);
483  vtxfit->AddTrack(3, wvKmTrk2);
484  vtxfit->AddVertex(0, vxpar, 0, 1);
485  if(!vtxfit->Fit(0)) continue;
486  vtxfit->Swim(0);
487 
488  WTrackParameter wpip = vtxfit->wtrk(0);
489  WTrackParameter wKp = vtxfit->wtrk(1);
490  WTrackParameter wKm1 = vtxfit->wtrk(2);
491  WTrackParameter wKm2 = vtxfit->wtrk(3);
492 
493  // * Get Kalman kinematic fit for this combination and store if better than previous one
494  double chisq = 999999.;
495  KalmanKinematicFit *kkmfit = KalmanKinematicFit::instance();
496  kkmfit->init();
497  kkmfit->AddTrack(0, wKm1); // K- (1st occurrence)
498  kkmfit->AddTrack(1, wpip); // pi+
499  kkmfit->AddTrack(2, wKm2); // K- (2nd occurrence)
500  kkmfit->AddTrack(3, wKp); // K+
501  kkmfit->AddFourMomentum(0, ecms); // 4 constraints: CMS energy and momentum
502  if(kkmfit->Fit()) {
503  chisq = kkmfit->chisq();
504  if(chisq < fSmallestChiSq) {
505  bestKalmanFit = kkmfit;
506  fSmallestChiSq = chisq;
507  }
508  }
509 
510  } // end of loop over pi^+
511  } // end of loop over K^+
512  } // end of loop over second K^-
513  } // end of loop over first K^-
514 
515  // * WRITE D0 and phi information from 4-constraint fit ("fit4c" branch)
516  if(fDoFit4c && bestKalmanFit && fSmallestChiSq < fMaxChiSq) {
517  HepLorentzVector pD0 = bestKalmanFit->pfit(0) + bestKalmanFit->pfit(1);
518  HepLorentzVector pphi = bestKalmanFit->pfit(2) + bestKalmanFit->pfit(3);
519  HepLorentzVector pJpsi = pD0 + pphi;
520  fFit4c["mD0"] = pD0.m();
521  fFit4c["mphi"] = pphi.m();
522  fFit4c["mJpsi"] = pJpsi.m();
523  fFit4c["chisq"] = fSmallestChiSq;
524  fNTupleMap["fit4c"]->write();
525  }
526 
527  } // end of fDoFit4c
528 
529 
530  // * STEP (G): WRITE Kalman 6-constraint kinematic fit with smallest chi squared ("fit6c" branch) * //
531  if(fDoFit6c) {
532 
533  // * Loop over all combinations and get the one with the smallest chi square
534  fSmallestChiSq = 9999999.;
535  KalmanKinematicFit *bestKalmanFit = KalmanKinematicFit::instance();
536  for(fKaonNeg1Iter = fKaonNeg.begin(); fKaonNeg1Iter != fKaonNeg.end(); ++fKaonNeg1Iter) {
537  for(fKaonNeg2Iter = fKaonNeg1Iter+1; fKaonNeg2Iter != fKaonNeg.end(); ++fKaonNeg2Iter) {

601  } // end of loop over second K^-
602  } // end of loop over first K^-
