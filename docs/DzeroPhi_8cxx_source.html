222  StatusCode DzeroPhi::execute()
223  {
224  // * STEP (A): Load next event from DST file * //
225 
226  // * Load event info *
227  /*
228  http://bes3.to.infn.it/Boss/7.0.2/html/namespaceEventModel_1_1EvtRec.html (namespace)
229  http://bes3.to.infn.it/Boss/7.0.2/html/classEvtRecEvent.html (class)
230  http://bes3.to.infn.it/Boss/7.0.2/html/EvtRecTrack_8h.html (typedef EvtRecTrackCol)
231  */
232  fLog << MSG::DEBUG << "Loading EventHeader, EvtRecEvent, and EvtRecTrackCol" << endreq;
233  fEventHeader = SmartDataPtr<Event::EventHeader>(eventSvc(), "/Event/EventHeader");
234  fEvtRecEvent = SmartDataPtr<EvtRecEvent> (eventSvc(), EventModel::EvtRec::EvtRecEvent);
235  fEvtRecTrkCol = SmartDataPtr<EvtRecTrackCol> (eventSvc(), EventModel::EvtRec::EvtRecTrackCol);
236 
237  // * Log run number, event number, and number of events *
238  fEvent_Ntotal = fEvtRecEvent->totalTracks();
239  fEvent_Ncharge = fEvtRecEvent->totalCharged();
240  fEvent_Nneutral = fEvtRecEvent->totalNeutral();
241  fLog << MSG::DEBUG
242  << "RUN " << fEventHeader->runNumber() << ", "
243  << "event number " << fEventHeader->eventNumber() << endreq;
244  fLog << MSG::DEBUG
245  << "Ncharged = " << fEvent_Ncharge << ", "
246  << "Nneutral = " << fEvent_Nneutral << ", "
247  << "Ntotal = " << fEvent_Ntotal << endreq;
248 
249  // * Set vertex origin *
250  IVertexDbSvc* vtxsvc;
251  Gaudi::svcLocator()->service("VertexDbSvc", vtxsvc);
252  if(vtxsvc->isVertexValid()){
253  double* dbv = vtxsvc->PrimaryVertex();
254  double* vv = vtxsvc->SigmaPrimaryVertex();
255  fEvent_Vx0 = dbv[0];
256  fEvent_Vy0 = dbv[1];
257  fEvent_Vz0 = dbv[2];
258  }
259 
260 
261  // * STEP (B): Create selection charged tracks * //
262 
263  // * Print log and set counters *
264  fLog << MSG::DEBUG << "Starting 'good' charged track selection:" << endreq;
265  fEvent_Nmdc = 0; // @todo Check if this makes sense at all
266  ParticleID *pid = ParticleID::instance();
267 
268  // * Clear vectors of selected particles *
269  fGoodChargedTracks.clear();
270  fKaonNeg.clear();
271  fKaonPos.clear();
272  fPionPos.clear();
273 
274  // * Loop over charged tracks *
275  for(int i = 0; i < fEvtRecEvent->totalCharged(); ++i) {
276  // Note: the first part of the set of reconstructed tracks are the charged tracks
277 
278  // * STEP 1: Get MDC information
279 
280  // * Get track info from Main Drift Chamber
281  fLog << MSG::DEBUG << " charged track " << i << "/" << fEvtRecEvent->totalCharged() << endreq;
282  fTrackIterator = fEvtRecTrkCol->begin() + i;
283  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
284  RecMdcTrack *mdcTrk = (*fTrackIterator)->mdcTrack();
285 
286  // * Get kinematics of track
287  fTrack_p = mdcTrk->p();
288  fTrack_Vx = mdcTrk->x();
289  fTrack_Vy = mdcTrk->y();
290  fTrack_Vz = mdcTrk->z();
291  fTrack_phi = mdcTrk->helix(1);
292  fTrack_Vr =
293  (fTrack_Vx - fEvent_Vx0) * cos(fTrack_phi) +
294  (fTrack_Vy - fEvent_Vy0) * sin(fTrack_phi);
295 
296  // * Get radii of track vertex
297  HepVector a = mdcTrk->helix();
298  HepSymMatrix Ea = mdcTrk->err();
299  HepPoint3D point0(0., 0., 0.); // initial point for MDC reconstruction
300  HepPoint3D IP(fEvent_Vx0, fEvent_Vy0, fEvent_Vz0);
301  VFHelix helixip(point0, a, Ea);
302  helixip.pivot(IP);
303  HepVector vecipa = helixip.a();
304  fTrack_Rvxy = fabs(vecipa[0]); // nearest distance to IP in xy plane
305  fTrack_Rvz = vecipa[3]; // nearest distance to IP in z direction
306  fTrack_Rvphi = vecipa[1]; // angle in the xy-plane (?)
307 
308  // * STEP 2: Apply vertex cuts, store track, and write info
309 
310  // * Apply vertex cuts
311  if(fabs(fTrack_Vz) >= fVz0cut) continue;
312  if(fabs(fTrack_Vr) >= fVr0cut) continue;
313  if(fabs(fTrack_Rvz) >= fRvz0cut) continue;
314  if(fabs(fTrack_Rvxy) >= fRvxy0cut) continue;
315 
316  // * Add charged track to vector
317  fGoodChargedTracks.push_back(*fTrackIterator);
318  fEvent_Nmdc += mdcTrk->charge(); // @todo Check if this makes sense at all
319 
320  // * WRITE primary vertex position info ("vxyz" branch)
321  if(fDoVxyz) fNTupleMap["vxyz"]->write();
322 
323  // * STEP 3: Distinguish and store types of particles
324 
325  // * Initialise PID sub-system
326  // if(pid) delete pid;
327  pid->init();
328  pid->setMethod(pid->methodProbability());
329  // pid->setMethod(pid->methodLikelihood()); // for likelihood Method
330  pid->setChiMinCut(4);
331  pid->setRecTrack(*fTrackIterator);
332  pid->usePidSys(pid->useDedx() | pid->useTof1() | pid->useTof2() | pid->useTofE()); // use PID sub-system
333  pid->identify(pid->onlyPion() | pid->onlyKaon()); // seperater Pion/Kaon
334  pid->calculate();
335  if(!(pid->IsPidInfoValid())) continue;
336 
337  // * WRITE particle identification info ("pid" branch)
338  if(fDoPID) {
339  fTrack_p = mdcTrk->p();
340  fPID_Cost = cos(mdcTrk->theta());
341  fPID_Dedx = pid->chiDedx(2);
342  fPID_Tof1 = pid->chiTof1(2);
343  fPID_Tof2 = pid->chiTof2(2);
344  fPID_Prob = pid->probPion();
345  fNTupleMap["pid"]->write();
346  }
347 
348  // * Identify type of charged particle and add to related vector (this package: kaon and pion) *
349  RecMdcKalTrack* mdcKalTrk = (*fTrackIterator)->mdcKalTrack(); // after ParticleID, use RecMdcKalTrack substitute RecMdcTrack
350  if(pid->probPion() > pid->probKaon()) { // NOTE: if more likely to be pion
351  if(pid->probPion() < 0.001) continue; // check if indeed a pion
352  if(pid->pdf(2) < pid->pdf(3)) continue; // for likelihood method (0=electron 1=muon 2=pion 3=kaon 4=proton)
353  RecMdcKalTrack::setPidType(RecMdcKalTrack::pion); // PID can be set to electron, muon, pion (default), kaon and proton
354  if(mdcKalTrk->charge() > 0) fPionPos.push_back(*fTrackIterator); // if positive pion
355  }
356  else { // NOTE: if more likely to be kaon (K+-)
357  if(pid->probKaon() < 0.001) continue; // check if indeed a kaon
358  if(pid->pdf(3) < pid->pdf(2)) continue; // for likelihood method (0=electron 1=muon 2=pion 3=kaon 4=proton)
359  RecMdcKalTrack::setPidType(RecMdcKalTrack::kaon); // PID can be set to electron, muon, pion (default), kaon and proton
360  if(mdcKalTrk->charge() < 0) fKaonNeg.push_back(*fTrackIterator); // if negative kaon
361  else fKaonPos.push_back(*fTrackIterator); // if positive kaon
362  }
363 
364  }
365 
366 
367  // * STEP (C): WRITE event info ("mult" branch) * //
368  fLog << MSG::DEBUG << "ngood, totcharge = " << fGoodChargedTracks.size() << " , " << fEvent_Nmdc << endreq;
369  if(fDoMult) {
370  fEvent_Ngood = fGoodChargedTracks.size();
371  fEvent_NKaonNeg = fKaonNeg.size();
372  fEvent_NKaonPos = fKaonPos.size();
373  fEvent_NPionPos = fPionPos.size();
374  fNTupleMap["mult"]->write();
375  } // end of fDoVxyz
376 
377 
378  // * STEP (D): WRITE dE/dx PID information ("dedx" branch) * //
379  if(fDoDedx) WriteDedxInfoForVector(fGoodChargedTracks, "dedx");
380  if(fDoDedx_Kaon) WriteDedxInfoForVector(fGoodChargedTracks, "dedx_K");
381  if(fDoDedx_Kaon) WriteDedxInfoForVector(fGoodChargedTracks, "dedx_K");
382  if(fDoDedx_Pion) WriteDedxInfoForVector(fGoodChargedTracks, "dedx_pi");
383 
384 
385  // * STEP (E): WRITE Time-of-Flight PID information ("tof*" branch) * //
386  if(fDoTofEC || fDoTofIB || fDoTofOB) for(fTrackIterator = fGoodChargedTracks.begin(); fTrackIterator != fGoodChargedTracks.end(); ++fTrackIterator) {
387 
388  // * Check if MDC and TOF info for track are valid *
389  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
390  if(!(*fTrackIterator)->isTofTrackValid()) continue;
391 
392  // * Get momentum as determined by MDC *
393  RecMdcTrack * mdcTrk = (*fTrackIterator)->mdcTrack();
394  double ptrk = mdcTrk->p();
395  SmartRefVector<RecTofTrack> tofTrkCol = (*fTrackIterator)->tofTrack();
396  SmartRefVector<RecTofTrack>::iterator iter_tof = tofTrkCol.begin();
397  for(; iter_tof != tofTrkCol.end(); ++iter_tof) {
398  TofHitStatus hitStatus;
399  hitStatus.setStatus((*iter_tof)->status());
400  if(!hitStatus.is_counter()) continue;
401  if(hitStatus.is_barrel()) {
402  if(hitStatus.layer() == 1) { // inner barrel
403  if(fDoTofIB) WriteTofInformation(iter_tof, ptrk, "tof1");
404  } else if(hitStatus.layer() == 2) { // outer barrel
405  if(fDoTofOB) WriteTofInformation(iter_tof, ptrk, "tof2");
406  }
407  }
408  else if(fDoTofEC && hitStatus.layer() == 0) // end cap
409  WriteTofInformation(iter_tof, ptrk, "tofe");
410  }
411 
412  } // loop all charged tracks
413 
414 
415  // * STEP (F): WRITE Kalman 4-constraint kinematic fit with smallest chi squared ("fit4c" branch) * //
416  if(fDoFit4c) {
417 
418  // * Loop over all combinations and get the one with the smallest chi square
419  fSmallestChiSq = 9999999.;
420  KalmanKinematicFit *bestKalmanFit = KalmanKinematicFit::instance();
421  for(fKaonNeg1Iter = fKaonNeg.begin(); fKaonNeg1Iter != fKaonNeg.end(); ++fKaonNeg1Iter) {
422  for(fKaonNeg2Iter = fKaonNeg1Iter+1; fKaonNeg2Iter != fKaonNeg.end(); ++fKaonNeg2Iter) {
423  for(fKaonPosIter = fKaonPos.begin(); fKaonPosIter != fKaonPos.end(); ++fKaonPosIter) {
424  for(fPionPosIter = fPionPos.begin(); fPionPosIter != fPionPos.end(); ++fPionPosIter) {
425 
426  // * Get Kalman tracks
427  RecMdcKalTrack *KmTrk1 = (*fKaonNeg1Iter)->mdcKalTrack();
428  RecMdcKalTrack *KmTrk2 = (*fKaonNeg2Iter)->mdcKalTrack();
429  RecMdcKalTrack *KpTrk = (*fKaonPosIter )->mdcKalTrack();
430  RecMdcKalTrack *pipTrk = (*fPionPosIter )->mdcKalTrack();
431 
432  // * Get W-tracks
433  WTrackParameter wvKmTrk1(mK, KmTrk1->getZHelix(), KmTrk1->getZError());
434  WTrackParameter wvKmTrk2(mK, KmTrk2->getZHelix(), KmTrk2->getZError());
435  WTrackParameter wvKpTrk (mK, KpTrk->getZHelix(), KpTrk->getZError());
436  WTrackParameter wvpipTrk(mpi, pipTrk->getZHelix(), pipTrk->getZError());
437 
438  // * Test vertex fit * //
439  HepPoint3D vx(0., 0., 0.);
440  HepSymMatrix Evx(3, 0);
441  double bx = 1E+6;
442  double by = 1E+6;
443  double bz = 1E+6;
444  Evx[0][0] = bx*bx;
445  Evx[1][1] = by*by;
446  Evx[2][2] = bz*bz;
447 
448  VertexParameter vxpar;
449  vxpar.setVx(vx);
450  vxpar.setEvx(Evx);
451 
452  VertexFit* vtxfit = VertexFit::instance();
453  vtxfit->init();
454  vtxfit->AddTrack(0, wvpipTrk);
455  vtxfit->AddTrack(1, wvKpTrk);
456  vtxfit->AddTrack(2, wvKmTrk1);
457  vtxfit->AddTrack(3, wvKmTrk2);
458  vtxfit->AddVertex(0, vxpar, 0, 1);
459  if(!vtxfit->Fit(0)) continue;
460  vtxfit->Swim(0);
461 
462  WTrackParameter wpip = vtxfit->wtrk(0);
463  WTrackParameter wKp = vtxfit->wtrk(1);
464  WTrackParameter wKm1 = vtxfit->wtrk(2);
465  WTrackParameter wKm2 = vtxfit->wtrk(3);
466 
467  // * Get Kalman kinematic fit for this combination and store if better than previous one
468  double chisq = 999999.;
469  KalmanKinematicFit *kkmfit = KalmanKinematicFit::instance();
470  kkmfit->init();
471  kkmfit->AddTrack(0, wKm1); // K- (1st occurrence)
472  kkmfit->AddTrack(1, wpip); // pi+
473  kkmfit->AddTrack(2, wKm2); // K- (2nd occurrence)
474  kkmfit->AddTrack(3, wKp); // K+
475  kkmfit->AddFourMomentum(0, ecms); // 4 constraints: CMS energy and momentum
476  if(kkmfit->Fit()) {
477  chisq = kkmfit->chisq();
478  if(chisq < fSmallestChiSq) {
479  bestKalmanFit = kkmfit;
480  fSmallestChiSq = chisq;
481  }
482  }
483 
484  } // end of loop over pi^+
485  } // end of loop over K^+
486  } // end of loop over second K^-
487  } // end of loop over first K^-
488 
489  // * WRITE D0 and phi information from 4-constraint fit ("fit4c" branch)
490  if(fDoFit4c && bestKalmanFit && fSmallestChiSq < fMaxChiSq) {
491  HepLorentzVector pD0 = bestKalmanFit->pfit(0) + bestKalmanFit->pfit(1);
492  HepLorentzVector pphi = bestKalmanFit->pfit(2) + bestKalmanFit->pfit(3);
493  HepLorentzVector pJpsi = pD0 + pphi;
494  fInvMass_D0 = pD0.m(); // invariant D0 mass according to Kalman kinematic fit
495  fInvMass_phi = pphi.m(); // invariant phi mass according to Kalman kinematic fit
496  fInvMass_Jpsi = pJpsi.m(); // invariant Jpsi mass according to Kalman kinematic fit
497  fInvMass_ChiSq = fSmallestChiSq; // chi square of the Kalman kinematic fit
498  fNTupleMap["fit4c"]->write();
499  }
500 
501  } // end of fDoFit4c
502 
503 
504  // * STEP (G): WRITE Kalman 6-constraint kinematic fit with smallest chi squared ("fit6c" branch) * //
505  if(fDoFit6c) {
506 
507  // * Loop over all combinations and get the one with the smallest chi square
508  fSmallestChiSq = 9999999.;
509  KalmanKinematicFit *bestKalmanFit = KalmanKinematicFit::instance();
510  for(fKaonNeg1Iter = fKaonNeg.begin(); fKaonNeg1Iter != fKaonNeg.end(); ++fKaonNeg1Iter) {

545  vtxfit->AddTrack(2, wvKmTrk1);

550

552  WTrackParameter wKp = vtxfit->wtrk(1);
553  WTrackParameter wKm1 = vtxfit->wtrk(2);
554  WTrackParameter wKm2 = vtxfit->wtrk(3);

559  kkmfit->init();

564  kkmfit->AddFourMomentum(2, ecms); // 4 constraints: CMS energy and momentum

569  fSmallestChiSq = chisq;
570  }
571  }
572 
573  } // end of loop over pi^+
574  } // end of loop over K^+
575  } // end of loop over second K^-
576  } // end of loop over first K^-
577

581  HepLorentzVector pphi = bestKalmanFit->pfit(2) + bestKalmanFit->pfit(3);

586  fInvMass_ChiSq = fSmallestChiSq;
587  fNTupleMap["fit6c"]->write();
588  }
589  } // end of fDoFit6c

592  return StatusCode::SUCCESS;
