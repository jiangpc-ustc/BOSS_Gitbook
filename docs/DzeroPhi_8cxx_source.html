251 StatusCode DzeroPhi::execute() {
252  fLog << MSG::INFO << "In execute():" << endreq;
253 
254  // * Load next event from DST file * //
255 
256  // * Load event info *
257  /*
258  http://bes3.to.infn.it/Boss/7.0.2/html/namespaceEventModel_1_1EvtRec.html (namespace)
259  http://bes3.to.infn.it/Boss/7.0.2/html/classEvtRecEvent.html (class)
260  http://bes3.to.infn.it/Boss/7.0.2/html/EvtRecTrack_8h.html (typedef EvtRecTrackCol)
261  */
262  fLog << MSG::DEBUG << "Loading EventHeader, EvtRecEvent, and EvtRecTrackCol" << endreq;
263  SmartDataPtr<Event::EventHeader> eventHeader (eventSvc(), "/Event/EventHeader");
264  SmartDataPtr<EvtRecEvent> evtRecEvent (eventSvc(), EventModel::EvtRec::EvtRecEvent);
265  SmartDataPtr<EvtRecTrackCol> evtRecTrkCol(eventSvc(), EventModel::EvtRec::EvtRecTrackCol);
266  fLog << MSG::DEBUG
267  << "eventHeader = " << eventHeader.ptr()
268  << ", evtRecEvent = " << evtRecEvent.ptr()
269  << ", evtRecTrkCol = " << evtRecTrkCol.ptr()
270  << endreq;
271 
272  // * Log run number, event number, and number of events *
273  fEvent_Ntotal = evtRecEvent->totalTracks();
274  fEvent_Ncharge = evtRecEvent->totalCharged();
275  fEvent_Nneutral = evtRecEvent->totalNeutral();
276  fLog << MSG::DEBUG
277  << "RUN " << eventHeader->runNumber() << ", "
278  << "event number " << eventHeader->eventNumber() << endreq;
279  fLog << MSG::DEBUG
280  << "Ncharged = " << fEvent_Ncharge << ", "
281  << "Nneutral = " << fEvent_Nneutral << ", "
282  << "Ntotal = " << fEvent_Ntotal << endreq;
283 
284 
285  // * Set vertex origin * //
286  IVertexDbSvc* vtxsvc;
287  Gaudi::svcLocator()->service("VertexDbSvc", vtxsvc);
288  if(vtxsvc->isVertexValid()){
289  double* dbv = vtxsvc->PrimaryVertex();
290  double* vv = vtxsvc->SigmaPrimaryVertex();
291  fEvent_Vx0 = dbv[0];
292  fEvent_Vy0 = dbv[1];
293  fEvent_Vz0 = dbv[2];
294  }
295 
296 
297  // * LOOP OVER CHARGED TRACKS: select 'good' charged tracks * //
298  fLog << MSG::DEBUG << "Starting 'good' charged track selection:" << endreq;
299  fEvent_Nmdc = 0; // @todo Check if this makes sense at all
300  fGoodChargedTracks.clear();
301  for(int i = 0; i < evtRecEvent->totalCharged(); ++i) {
302  // Note: the first part of the set of reconstructed tracks are the charged tracks
303 
304  // * Check if valid according to MDC
305  fLog << MSG::DEBUG << " charged track " << i << "/" << evtRecEvent->totalCharged() << endreq;
306  fTrackIterator = evtRecTrkCol->begin() + i;
307  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
308 
309  // * Get track info from Main Drift Chamber
310  RecMdcTrack *mdcTrk = (*fTrackIterator)->mdcTrack();
311  fTrack_p = mdcTrk->p();
312  fTrack_Vx = mdcTrk->x();
313  fTrack_Vy = mdcTrk->y();
314  fTrack_Vz = mdcTrk->z();
315  fTrack_phi = mdcTrk->helix(1);
316  fTrack_Vr =
317  (fTrack_Vx - fEvent_Vx0) * cos(fTrack_phi) +
318  (fTrack_Vy - fEvent_Vy0) * sin(fTrack_phi);
319 
320  // * Get radii of vertex
321  HepVector a = mdcTrk->helix();
322  HepSymMatrix Ea = mdcTrk->err();
323  HepPoint3D point0(0., 0., 0.); // initial point for MDC reconstruction
324  HepPoint3D IP(fEvent_Vx0, fEvent_Vy0, fEvent_Vz0);
325  VFHelix helixip(point0, a, Ea);
326  helixip.pivot(IP);
327  HepVector vecipa = helixip.a();
328  fTrack_Rvxy = fabs(vecipa[0]); // nearest distance to IP in xy plane
329  fTrack_Rvz = vecipa[3]; // nearest distance to IP in z direction
330  fTrack_Rvphi = vecipa[1]; // angle in the xy-plane (?)
331 
332  // * Apply vertex cuts
333  if(fabs(fTrack_Vz) >= fVz0cut) continue;
334  if(fabs(fTrack_Vr) >= fVr0cut) continue;
335  if(fabs(fTrack_Rvz) >= fRvz0cut) continue;
336  if(fabs(fTrack_Rvxy) >= fRvxy0cut) continue;
337 
338  // * Add charged track to vector
339  fGoodChargedTracks.push_back(*fTrackIterator);
340  fEvent_Nmdc += mdcTrk->charge(); // @todo Check if this makes sense at all
341 
342  // * WRITE primary vertex position info ("vxyz" branch)
343  if(fDoVxyz) fNTupleMap["vxyz"]->write();
344 
345  }
346 
347 
348  // * Store dE/dx PID information * //
349  if(fDoDedx) {
350  fTrackIterator = fGoodChargedTracks.begin();
351  for(; fTrackIterator != fGoodChargedTracks.end(); ++fTrackIterator) {
352 
353  // * Check if dE/dx and MDC info exists *
354  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
355  if(!(*fTrackIterator)->isMdcDedxValid()) continue;
356  RecMdcTrack* mdcTrk = (*fTrackIterator)->mdcTrack();
357  RecMdcDedx* dedxTrk = (*fTrackIterator)->mdcDedx();
358 
359  // * WRITE energy loss PID info ("dedx" branch) *
360  fTrack_p = mdcTrk->p();
361  fChi2e = dedxTrk->chiE();
362  fChi2mu = dedxTrk->chiMu();
363  fChi2pi = dedxTrk->chiPi();
364  fChi2k = dedxTrk->chiK();
365  fChi2p = dedxTrk->chiP();
366  fProbPH = dedxTrk->probPH();
367  fNormPH = dedxTrk->normPH();
368  fGhit = dedxTrk->numGoodHits();
369  fThit = dedxTrk->numTotalHits();
370  fNTupleMap["dedx"]->write();
371 
372  } // end of loop over all charged track
373  } // end of fDoDedx
374 
375 
376  // * Store Time-of-Flight information * //
377  if(fDoTofEC || fDoTofIB || fDoTofOB) {
378  fTrackIterator = fGoodChargedTracks.begin();
379  for(; fTrackIterator != fGoodChargedTracks.end(); ++fTrackIterator) {
380 
381  // * Check if MDC and TOF info for track are valid *
382  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
383  if(!(*fTrackIterator)->isTofTrackValid()) continue;
384 
385  // * Get momentum as determined by MDC *
386  RecMdcTrack * mdcTrk = (*fTrackIterator)->mdcTrack();
387  double ptrk = mdcTrk->p();
388  SmartRefVector<RecTofTrack> tofTrkCol = (*fTrackIterator)->tofTrack();
389  SmartRefVector<RecTofTrack>::iterator iter_tof = tofTrkCol.begin();
390  for(; iter_tof != tofTrkCol.end(); ++iter_tof) {
391  TofHitStatus hitStatus;
392  hitStatus.setStatus((*iter_tof)->status());
393  if(!hitStatus.is_counter()) continue;
394  if(hitStatus.is_barrel()) {
395  if(hitStatus.layer() == 1) { // inner barrel
396  if(fDoTofIB) GetTofInformation(iter_tof, ptrk, "tof1");
397  } else if(hitStatus.layer() == 2) { // outer barrel
398  if(fDoTofOB) GetTofInformation(iter_tof, ptrk, "tof2");
399  }
400  }
401  else if(fDoTofEC && hitStatus.layer() == 0) // end cap
402  GetTofInformation(iter_tof, ptrk, "tofe");
403  }
404 
405  } // loop all charged track
406  } // end of fDoTofEC, fDoTofIB, fDoTofOB
407 
408 
409  // * Create a selection of charged decay products * //
410 
411  // * Clear lists of decay products and initialise PID instance *
412  fKaonNeg.clear();
413  fKaonPos.clear();
414  fPionPos.clear();
415  ParticleID *pid = ParticleID::instance();
416 
417  // * Loop over collection of 'good' charged tracks *
418  fTrackIterator = fGoodChargedTracks.begin();
419  for(; fTrackIterator != fGoodChargedTracks.end(); ++fTrackIterator) {
420 
421  // * Initialise PID sub-system
422  // if(pid) delete pid;
423  pid->init();
424  pid->setMethod(pid->methodProbability());
425  // pid->setMethod(pid->methodLikelihood()); // for likelihood Method
426  pid->setChiMinCut(4);
427  pid->setRecTrack(*fTrackIterator);
428  pid->usePidSys(pid->useDedx() | pid->useTof1() | pid->useTof2() | pid->useTofE()); // use PID sub-system
429  pid->identify(pid->onlyPion() | pid->onlyKaon()); // seperater Pion/Kaon
430  pid->calculate();
431  if(!(pid->IsPidInfoValid())) continue;
432 
433  // * Get MDC track info *
434  RecMdcTrack* mdcTrk = (*fTrackIterator)->mdcTrack();
435 
436  // * WRITE particle identification info ("pid" branch) *
437  if(fDoPID) {
438  fTrack_p = mdcTrk->p();
439  fCostPID = cos(mdcTrk->theta());
440  fDedxPID = pid->chiDedx(2);
441  fTof1PID = pid->chiTof1(2);
442  fTof2PID = pid->chiTof2(2);
443  fProbPID = pid->probPion();
444  fNTupleMap["pid"]->write();
445  }
446 
447  // * If more likely to be pion (pi+-) *
448  if(pid->probPion() > pid->probKaon()) {
449 
450  // * Check if indeed a pion
451  if(pid->probPion() < 0.001) continue;
452  if(pid->pdf(2) < pid->pdf(3)) continue;
453  // for likelihood method (0=electron 1=muon 2=pion 3=kaon 4=proton)
454 
455  // * Get MDC Kalman track
456  RecMdcKalTrack* mdcKalTrk = (*fTrackIterator)->mdcKalTrack();
457  // after ParticleID, use RecMdcKalTrack substitute RecMdcTrack
458  RecMdcKalTrack::setPidType(RecMdcKalTrack::pion);
459  // PID can set to electron, muon, pion (default), kaon and proton
460 
461  // * If positive pion
462  if(mdcKalTrk->charge() > 0)
463  fPionPos.push_back(*fTrackIterator);
464 
465  }
466 
467  // * If more likely to be kaon (K+-) *
468  else {
469 
470  // * Check if indeed a kaon
471  if(pid->probKaon() < 0.001) continue;
472  if(pid->pdf(3) < pid->pdf(2)) continue;
473  // for likelihood method (0=electron 1=muon 2=pion 3=kaon 4=proton)
474 
475  // * Get MDC Kalman track
476  RecMdcKalTrack* mdcKalTrk = (*fTrackIterator)->mdcKalTrack();
477  // after ParticleID, use RecMdcKalTrack substitute RecMdcTrack
478  RecMdcKalTrack::setPidType(RecMdcKalTrack::kaon);
479  // PID can set to electron, muon, pion (default), kaon and proton
480 
481  // * If negative kaon
482  if(mdcKalTrk->charge() < 0)
483  fKaonNeg.push_back(*fTrackIterator);
484 
485  // * If positive kaon
486  else
487  fKaonPos.push_back(*fTrackIterator);
488 
489  }
490  }
491 
492  // * WRITE event info ("mult" branch) * //
493  fLog << MSG::DEBUG << "ngood, totcharge = " << fGoodChargedTracks.size() << " , " << fEvent_Nmdc << endreq;
494  if(fDoMult) {
495  fEvent_Ngood = fGoodChargedTracks.size();
496  fEvent_NKaonNeg = fKaonNeg.size();
497  fEvent_NKaonPos = fKaonPos.size();
498  fEvent_NPionPos = fPionPos.size();
499  fNTupleMap["mult"]->write();
500  } // end of fDoVxyz
501 
502 
503  // ! Cut on number of particles ! //
504  // @todo Test if procedure can be made to work for arbitrary number of kaons and pions
505  if(fKaonNeg.size() != 2) return SUCCESS;
506  if(fKaonPos.size() != 1) return SUCCESS;
507  if(fPionPos.size() != 1) return SUCCESS;
508 
509 
510  // * Test all combinations with K- and store the ones closes to D0 and to phi * //
511 
512  // * Get Kalman tracks
513  RecMdcKalTrack *pipTrk = (*(fPionPos.begin())) ->mdcKalTrack();
514  RecMdcKalTrack *KpTrk = (*(fKaonPos.begin())) ->mdcKalTrack();
515  RecMdcKalTrack *KmTrk1 = (*(fKaonNeg.begin())) ->mdcKalTrack();
516  RecMdcKalTrack *KmTrk2 = (*(fKaonNeg.begin()+1))->mdcKalTrack();
517 
518  // * Get W-tracks
519  WTrackParameter wvpipTrk(mpi, pipTrk->getZHelix(), pipTrk->getZError());
520  WTrackParameter wvKpTrk (mK, KpTrk->getZHelix(), KpTrk->getZError());
521  WTrackParameter wvKmTrk1(mK, KmTrk1->getZHelix(), KmTrk1->getZError());
522  WTrackParameter wvKmTrk2(mK, KmTrk2->getZHelix(), KmTrk2->getZError());
523 
524 
525  // * Test vertex fit * //
526  HepPoint3D vx(0., 0., 0.);
527  HepSymMatrix Evx(3, 0);
528  double bx = 1E+6;
529  double by = 1E+6;
530  double bz = 1E+6;
531  Evx[0][0] = bx*bx;
532  Evx[1][1] = by*by;
533  Evx[2][2] = bz*bz;
534 
535  VertexParameter vxpar;
536  vxpar.setVx(vx);
537  vxpar.setEvx(Evx);
538 
539  VertexFit* vtxfit = VertexFit::instance();
540  vtxfit->init();
541  vtxfit->AddTrack(0, wvpipTrk);
