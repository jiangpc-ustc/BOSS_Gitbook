307 StatusCode DzeroPhi::execute() {
308 
309  MsgStream log(msgSvc(), name());
310  log << MSG::INFO << "In execute():" << endreq;
311 
312  // * Load next event from DST file * //
313  // NOTE: Ncut0 -- no cut yet
314 
315  // * Load DST file info *
316  SmartDataPtr<Event::EventHeader> eventHeader(eventSvc(), "/Event/EventHeader");
317  int runNo = eventHeader->runNumber();
318  int evtNo = eventHeader->eventNumber();
319  log << MSG::DEBUG << "run, evtnum = "
320  << runNo << " , "
321  << evtNo << endreq;
322  cout << "event number: " << evtNo << endl;
323  Ncut0++; // counter for all events
324 
325  // * Load event information and track collection *
326  /*
327  http://bes3.to.infn.it/Boss/7.0.2/html/namespaceEventModel_1_1EvtRec.html (namespace)
328  http://bes3.to.infn.it/Boss/7.0.2/html/classEvtRecEvent.html (class)
329  http://bes3.to.infn.it/Boss/7.0.2/html/EvtRecTrack_8h.html (typedef EvtRecTrackCol)
330  */
331  SmartDataPtr<EvtRecEvent> evtRecEvent (eventSvc(), EventModel::EvtRec::EvtRecEvent);
332  SmartDataPtr<EvtRecTrackCol> evtRecTrkCol(eventSvc(), EventModel::EvtRec::EvtRecTrackCol);
333 
334  // * Log number of events *
335  log << MSG::DEBUG << "Ncharged, Nneutral, Ntotal = "
336  << evtRecEvent->totalCharged() << ", "
337  << evtRecEvent->totalNeutral() << ", "
338  << evtRecEvent->totalTracks() << endreq;
339 
340 
341  // * Set vertex origin * //
342  Hep3Vector xorigin(0,0,0);
343  IVertexDbSvc* vtxsvc;
344  Gaudi::svcLocator()->service("VertexDbSvc", vtxsvc);
345  if(vtxsvc->isVertexValid()){
346  double* dbv = vtxsvc->PrimaryVertex();
347  double* vv = vtxsvc->SigmaPrimaryVertex();
348  // HepVector dbv = fReader.PrimaryVertex(runNo);
349  // HepVector vv = fReader.SigmaPrimaryVertex(runNo);
350  xorigin.setX(dbv[0]);
351  xorigin.setY(dbv[1]);
352  xorigin.setZ(dbv[2]);
353  }
354 
355 
356  // * LOOP OVER CHARGED TRACKS: select 'good' charged tracks * //
357  // NOTE: Ncut1 -- vertex cuts
358  // The first part of the set of reconstructed tracks are the charged tracks
359  int nCharge = 0;
361  Vint iGood;
363  for(int i = 0; i < evtRecEvent->totalCharged(); ++i) {
364 
365  // * Get track: beginning of all tracks + event number
366  EvtRecTrackIterator itTrk = evtRecTrkCol->begin() + i;
367  if(!(*itTrk)->isMdcTrackValid()) continue;
368 
369  // * Get track info from Main Drift Chamber
370  RecMdcTrack *mdcTrk = (*itTrk)->mdcTrack();
371  double pch = mdcTrk->p();
372  double x0 = mdcTrk->x();
373  double y0 = mdcTrk->y();
374  double z0 = mdcTrk->z();
375  double phi0 = mdcTrk->helix(1);
376 
377  // * Get vertex origin
378  double Rxy =
379  (x0 - xorigin.x()) * cos(phi0) +
380  (y0 - xorigin.y()) * sin(phi0);
381 
382  // * Get radii of vertex
383  HepVector a = mdcTrk->helix();
384  HepSymMatrix Ea = mdcTrk->err();
385  HepPoint3D point0(0., 0., 0.); // the initial point for MDC recosntruction
386  HepPoint3D IP(xorigin[0], xorigin[1], xorigin[2]);
387  VFHelix helixip(point0, a, Ea);
388  helixip.pivot(IP);
389  HepVector vecipa = helixip.a();
390  double Rvxy0 = fabs(vecipa[0]); // nearest distance to IP in xy plane
391  double Rvz0 = vecipa[3]; // nearest distance to IP in z direction
392  double Rvphi0 = vecipa[1];
393 
394  // * WRITE primary vertex position info ("vxyz" branch) *
395  fVx0 = x0; // primary x-vertex as determined by MDC
396  fVy0 = y0; // primary y-vertex as determined by MDC
397  fVz0 = z0; // primary z-vertex as determined by MDC
398  fVr0 = Rxy; // distance from origin in xy-plane
399  fRvxy0 = Rvxy0; // nearest distance to IP in xy plane
400  fRvz0 = Rvz0; // nearest distance to IP in z direction
401  fRvphi0 = Rvphi0; // angle in the xy-plane (?)
402  fTupleVxyz->write(); // "vxyz" branch
403 
404  // * Apply vertex cuts *
405  if(fabs(z0) >= fVz0cut) continue;
406  if(fabs(Rxy) >= fVr0cut) continue;
407  if(fabs(Rvz0) >= fRvz0cut) continue;
408  if(fabs(Rvxy0) >= fRvxy0cut) continue;
409 
410  // * Add charged track to vector *
411  iGood.push_back(i);
412  nCharge += mdcTrk->charge();
413 
414  }
415 
416  // * Finish Good Charged Track Selection * //
417  log << MSG::DEBUG << "ngood, totcharge = " << iGood.size() << " , " << nCharge << endreq;
418  Ncut1++; // vertex cuts
419 
420 
421  // * Store dE/dx PID information * //
422  for(int i = 0; i < iGood.size(); ++i) {
423 
424  // * Get track *
425  EvtRecTrackIterator itTrk = evtRecTrkCol->begin() + iGood[i];
426  if(!(*itTrk)->isMdcTrackValid()) continue;
427  if(!(*itTrk)->isMdcDedxValid()) continue;
428  RecMdcTrack* mdcTrk = (*itTrk)->mdcTrack();
429  RecMdcDedx* dedxTrk = (*itTrk)->mdcDedx();
430 
431  // * WRITE energy loss PID info ("dedx" branch) *
432  fPtrack = mdcTrk->p(); // momentum of the track
433  fChi2e = dedxTrk->chiE(); // chi2 in case of electron
434  fChi2mu = dedxTrk->chiMu(); // chi2 in case of muon
435  fChi2pi = dedxTrk->chiPi(); // chi2 in case of pion
436  fChi2k = dedxTrk->chiK(); // chi2 in case of kaon
437  fChi2p = dedxTrk->chiP(); // chi2 in case of proton
438  fProbPH = dedxTrk->probPH(); // most probable pulse height from truncated mean
439  fNormPH = dedxTrk->normPH(); // normalized pulse height
440  fGhit = dedxTrk->numGoodHits(); // number of good hits
441  fThit = dedxTrk->numTotalHits(); // total number of hits
442  fTupleDedx->write(); // "dedx" branch
443  }
444 
445 
446  // * Store Time-of-Flight information * //
447  for(int i = 0; i < iGood.size(); ++i) {
448  EvtRecTrackIterator itTrk = evtRecTrkCol->begin() + iGood[i];
449  if(!(*itTrk)->isMdcTrackValid()) continue;
450  if(!(*itTrk)->isTofTrackValid()) continue;
451 
452  RecMdcTrack * mdcTrk = (*itTrk)->mdcTrack();
453  SmartRefVector<RecTofTrack> tofTrkCol = (*itTrk)->tofTrack();
454 
455  double ptrk = mdcTrk->p();
456 
457  SmartRefVector<RecTofTrack>::iterator iter_tof = tofTrkCol.begin();
458  for(;iter_tof != tofTrkCol.end(); ++iter_tof) {
459  TofHitStatus hitStatus;
460  hitStatus.setStatus((*iter_tof)->status());
461  if(!hitStatus.is_counter()) continue;
462 
463  // * If end cap ToF detector: *
464  if(!hitStatus.is_barrel()) { // is not barrel
465  if(hitStatus.layer() != 0) continue; // abort if not end cap
466 
467  // * Get ToF info *
468  double path = (*iter_tof)->path(); // distance of flight
469  double tof = (*iter_tof)->tof(); // time of flight
470  double ph = (*iter_tof)->ph(); // ToF pulse height
471  double rhit = (*iter_tof)->zrhit(); // Track extrapolate Z or R Hit position
472  double qual = 0.+(*iter_tof)->quality(); // data quality of reconstruction
473  double cntr = 0.+(*iter_tof)->tofID(); // ToF counter ID
474 
475  // * Get ToF for each particle hypothesis *
476  double texp[5];
477  for(int j = 0; j < 5; j++) {
478  double gb = ptrk/xmass[j]; // v = p/m (non-relativistic velocity)
479  double beta = gb/sqrt(1+gb*gb);
480  texp[j] = 10 * path /beta/velc_mm; // hypothesis ToF
481  }
482 
483  // * WRITE ToF end cap info ("tofe" branch) *
484  fPtotTofEC = ptrk; // momentum of the track as reconstructed by MDC
485  fPathTofEC = path; // path length
486  fTofEC = tof; // time of flight
487  fCntrTofEC = cntr; // ToF counter ID
488  fPhTofEC = ph; // ToF pulse height
489  fRhitTofEC = rhit; // track extrapolate Z or R Hit position
490  fQualTofEC = qual; // data quality of reconstruction
491  fElectronTofEC = tof - texp[0]; // difference with ToF in electron hypothesis
