151  int ConfigLoader::LoadConfiguration(const char* filename)
152  {
154  std::ifstream file(filename);
155  if(!file.is_open()) {
156  std::cout << "WARNING: Could not load configuration file \"" << filename << "\"" << std::endl;
157  return -1;
158  }
160  std::cout << std::endl << "LOADING CONFIGURATION FROM \"" << filename << "\"" << std::endl;
162  std::string line;
163  while(getline(file, line)) {
166  if(line.back()<' ') line.pop_back();
168  if(line.find('=') == std::string::npos && line.find(' ') == std::string::npos) continue;
170  RemoveLeading(line);
171  RemoveLeading(line, '\t');
173  std::string beginning = line.substr(0,2);
174  if(!beginning.compare("//")) continue; // C++ line comment
175  if(!beginning.compare("/*")) continue; // C++ begin block comment
176  if(beginning.front() == '*') continue; // C++ continue block comment
177  if(beginning.front() == '#') continue; // bash comment
178  if(beginning.front() == '%') continue; // LaTeX comment
180  std::string parname { GetParameterName (line) };
181  std::string parvalue { GetParameterValue(line) };
183  ArgPair_base::SetParameters(parname, parvalue);
185  }
187  ArgPair_base::PrintAll();
189  return ArgPair_base::instances.size();
190  }
