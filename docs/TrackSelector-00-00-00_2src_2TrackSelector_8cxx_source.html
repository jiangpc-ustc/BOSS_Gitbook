166  StatusCode TrackSelector::execute()
167  {
169 
170  // * Load event info *
171  /*
172  http://bes3.to.infn.it/Boss/7.0.2/html/namespaceEventModel_1_1EvtRec.html (namespace)
173  http://bes3.to.infn.it/Boss/7.0.2/html/classEvtRecEvent.html (class)
174  http://bes3.to.infn.it/Boss/7.0.2/html/EvtRecTrack_8h.html (typedef EvtRecTrackCol)
175  */
176  fLog << MSG::DEBUG << "Loading EventHeader, EvtRecEvent, and EvtRecTrackCol" << endreq;
177  fEventHeader = SmartDataPtr<Event::EventHeader>(eventSvc(), "/Event/EventHeader");
178  fEvtRecEvent = SmartDataPtr<EvtRecEvent> (eventSvc(), EventModel::EvtRec::EvtRecEvent);
179  fEvtRecTrkCol = SmartDataPtr<EvtRecTrackCol> (eventSvc(), EventModel::EvtRec::EvtRecTrackCol);
180 
181  // * Log run number, event number, and number of events *
182  fLog << MSG::DEBUG
183  << "RUN " << fEventHeader->runNumber() << ", "
184  << "event number " << fEventHeader->eventNumber() << endreq;
185  fLog << MSG::DEBUG
186  << "Ncharged = " << fEvtRecEvent->totalCharged() << ", "
187  << "Nneutral = " << fEvtRecEvent->totalNeutral() << ", "
188  << "Ntotal = " << fEvtRecEvent->totalTracks() << endreq;
189 
190  // * Set vertex origin *
191  IVertexDbSvc* vtxsvc;
192  Gaudi::svcLocator()->service("VertexDbSvc", vtxsvc);
193  double v0x, v0y, v0z;
194  if(vtxsvc->isVertexValid()){
195  double* dbv = vtxsvc->PrimaryVertex();
196  double* vv = vtxsvc->SigmaPrimaryVertex();
197  v0x = dbv[0];
198  v0y = dbv[1];
199  v0z = dbv[2];
200  }
201 
202 
204 
205  // * Print log and set counters *
206  fLog << MSG::DEBUG << "Starting 'good' charged track selection:" << endreq;
207  int nChargesMDC = 0;
208  ParticleID *pid = ParticleID::instance();
209 
210  // * Clear vectors of selected particles *
211  fGoodChargedTracks.clear();
212 
213  // * Loop over charged tracks *
214  for(int i = 0; i < fEvtRecEvent->totalCharged(); ++i) {
215  // Note: the first part of the set of reconstructed tracks are the charged tracks
217 
218  // * Get track info from Main Drift Chamber
219  fLog << MSG::DEBUG << " charged track " << i << "/" << fEvtRecEvent->totalCharged() << endreq;
220  fTrackIterator = fEvtRecTrkCol->begin() + i;
221  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
222  fTrackMDC = (*fTrackIterator)->mdcTrack();
223 
224  // * Get kinematics of track
225  double phi = fTrackMDC->helix(1);
226  double vr =
227  (fTrackMDC->x() - v0x) * cos(phi) +
228  (fTrackMDC->y() - v0y) * sin(phi);
229 
230  // * Get radii of track vertex
231  HepVector a = fTrackMDC->helix();
232  HepSymMatrix Ea = fTrackMDC->err();
233  HepPoint3D point0(0., 0., 0.); // initial point for MDC reconstruction
234  HepPoint3D IP(v0x, v0y, v0z);
235  VFHelix helixip(point0, a, Ea);
236  helixip.pivot(IP);
237  HepVector vecipa = helixip.a();
238  double rvxy = fabs(vecipa[0]); // nearest distance to IP in xy plane
239  double rvz = vecipa[3]; // nearest distance to IP in z direction
240  double rvphi = vecipa[1]; // angle in the xy-plane (?)
241 
243 
244  // * Apply vertex cuts
245  if(fTrackMDC->z() >= fVz0cut) continue;
246  if(vr >= fVr0cut) continue;
247  if(rvz >= fRvz0cut) continue;
248  if(rvxy >= fRvxy0cut) continue;
249 
250  // * Add charged track to vector
251  fGoodChargedTracks.push_back(*fTrackIterator);
252  nChargesMDC += fTrackMDC->charge(); // @todo Check if this makes sense at all
253 
255  if(fDoTrackVertex) {
256  fTrackVertex.at("vx") = fTrackMDC->x();
257  fTrackVertex.at("vy") = fTrackMDC->y();
258  fTrackVertex.at("vz") = fTrackMDC->z();
259  fTrackVertex.at("vr") = vr;
260  fTrackVertex.at("rvxy") = rvxy;
261  fTrackVertex.at("rvz") = rvz;
262  fTrackVertex.at("rvphi") = rvphi;
263  fTrackVertex.at("phi") = phi;
264  fTrackVertex.at("p") = fTrackMDC->p();
265  fNTupleMap.at("vxyz")->write();
266  }
267 
269  if(fDoDedx) WriteDedxInfo(*fTrackIterator, "dedx", fDedx);
270 
272  if(fDoTofEC || fDoTofIB || fDoTofOB) {
273 
274  // * Check if MDC and TOF info for track are valid *
275  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
276  if(!(*fTrackIterator)->isTofTrackValid()) continue;
277 
278  // * Get momentum as determined by MDC *
279  fTrackMDC = (*fTrackIterator)->mdcTrack();
280  double ptrk;
281  if(fTrackMDC) fTrackMDC->p();
282  SmartRefVector<RecTofTrack> tofTrkCol = (*fTrackIterator)->tofTrack();
283  SmartRefVector<RecTofTrack>::iterator iter_tof = tofTrkCol.begin();
284  for(; iter_tof != tofTrkCol.end(); ++iter_tof) {
285  TofHitStatus hitStatus;
286  hitStatus.setStatus((*iter_tof)->status());
287  if(!hitStatus.is_counter()) continue;
288  if(hitStatus.is_barrel()) {
289  if(hitStatus.layer() == 1) { // inner barrel
290  if(fDoTofIB) WriteTofInformation(iter_tof, ptrk, "tof1", fTofIB);
291  } else if(hitStatus.layer() == 2) { // outer barrel
292  if(fDoTofOB) WriteTofInformation(iter_tof, ptrk, "tof2", fTofOB);
293  }
294  }
295  else if(fDoTofEC && hitStatus.layer() == 0) // end cap
296  WriteTofInformation(iter_tof, ptrk, "tofe", fTofEC);
297  }
298 
299  } // if(fDoTofEC || fDoTofIB || fDoTofOB)
300 
301  }
302 
303 
305  fLog << MSG::DEBUG << "ngood, totcharge = " << fGoodChargedTracks.size() << " , " << nChargesMDC << endreq;
306  if(fDoMult) {
307  fMult.at("Ntotal") = fEvtRecEvent->totalTracks();
308  fMult.at("Ncharge") = fEvtRecEvent->totalCharged();
309  fMult.at("Nneutral") = fEvtRecEvent->totalNeutral();
310  fMult.at("Ngood") = fGoodChargedTracks.size();
311  fMult.at("Nmdc") = nChargesMDC;
312  fNTupleMap.at("vertex")->write();
313  }
314  if(fDoVertex) {
315  fVertex.at("vx0") = v0x;
316  fVertex.at("vy0") = v0y;
317  fVertex.at("vz0") = v0z;
318  fNTupleMap.at("vertex")->write();
319  }
320 
322  return execute_rest();
323  return StatusCode::SUCCESS;
324  }

409  template<typename TYPE>
410  void TrackSelector::WriteTofInformation(SmartRefVector<RecTofTrack>::iterator iter_tof, double ptrk, const char* tupleName, std::map<std::string, NTuple::Item<TYPE> > &map)
411  {
412 
413  // * Get ToF for each particle hypothesis
414  double path = (*iter_tof)->path();
415  std::vector<double> texp(gNMasses);
416  for(size_t j = 0; j < texp.size(); ++j) {
417  double gb = ptrk/gMasses[j]; // v = p/m (non-relativistic velocity)
418  double beta = gb/sqrt(1+gb*gb);
419  texp[j] = 10 * path /beta/gSpeedOfLight; // hypothesis ToF
420  }
421 
422  // * WRITE ToF info
423  map.at("ptrk") = ptrk;
424  map.at("tof") = (*iter_tof)->tof();
425  map.at("path") = (*iter_tof)->path();
426  map.at("cntr") = (*iter_tof)->tofID();
427  map.at("ph") = (*iter_tof)->ph();
428  map.at("zrhit") = (*iter_tof)->zrhit();
429  map.at("qual") = (*iter_tof)->quality();
430  map.at("te") = path - texp[0];
431  map.at("tmu") = path - texp[1];
432  map.at("tpi") = path - texp[2];
433  map.at("tk") = path - texp[3];
434  map.at("tp") = path - texp[4];
435  fNTupleMap.at(tupleName)->write();
436 
437  }
